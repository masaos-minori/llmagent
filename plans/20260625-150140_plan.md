## Design

**`test_eventbus_broker.py` — broker ユニットテスト (anyio):**

```python
import asyncio, pytest
from eventbus.broker import EventBroker

@pytest.mark.anyio
async def test_broker_fan_out():
    broker = EventBroker()
    sub1 = broker.subscribe([])
    sub2 = broker.subscribe([])
    broker.publish({"seq": 1, "topic": "t", "event_id": "a"})
    assert sub1.queue.qsize() == 1
    assert sub2.queue.qsize() == 1

@pytest.mark.anyio
async def test_broker_topic_filter():
    broker = EventBroker()
    sub_t = broker.subscribe(["t"])
    sub_all = broker.subscribe([])
    broker.publish({"seq": 1, "topic": "other"})
    assert sub_t.queue.empty()   # topic 不一致 → 届かない
    assert sub_all.queue.qsize() == 1

@pytest.mark.anyio
async def test_broker_unsubscribe():
    broker = EventBroker()
    sub = broker.subscribe([])
    broker.unsubscribe(sub)
    broker.publish({"seq": 1, "topic": "t"})
    assert sub.queue.empty()
```
