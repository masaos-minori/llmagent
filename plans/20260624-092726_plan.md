## Design

```python
# repl.py — SIGTERM handler registration (in __init__ or start())
import signal
loop = asyncio.get_event_loop()
loop.add_signal_handler(signal.SIGTERM, self._handle_sigterm)

def _handle_sigterm(self) -> None:
    if self._ctx.conv.is_processing:
        asyncio.ensure_future(self._graceful_shutdown())
    else:
        asyncio.ensure_future(self._immediate_shutdown())

async def _graceful_shutdown(self) -> None:
    try:
        await asyncio.wait_for(self._wait_for_turn_complete(), timeout=10.0)
    except asyncio.TimeoutError:
        pass  # timeout hit — force close
    await self._ctx.services.lifecycle.shutdown_all()
    sys.exit(0)
```
