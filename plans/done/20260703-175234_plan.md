## Design

### Test structure

```python
import asyncio
from unittest.mock import AsyncMock, MagicMock, patch
import pytest
import httpx

from shared.http_transport import TransportError
from shared.mcp_config import McpServerConfig, McpServerHealthState, StartupMode, TransportType
from shared.tool_transport_invoker import ToolTransportInvoker
from shared.transport_dto import ToolCallResult


def _cfg(url: str = "http://127.0.0.1:8000") -> McpServerConfig:
    return McpServerConfig(TransportType.HTTP, url, startup_mode=StartupMode.PERSISTENT)


def _ok() -> ToolCallResult:
    return ToolCallResult(output="ok", is_error=False, request_id="req", server_key="srv")


def _tool_err() -> ToolCallResult:
    return ToolCallResult(output="err", is_error=True, request_id="", server_key="srv", error_type="tool")


class TestToolTransportInvoker:

    def _make_invoker(self, server_key="srv", url="http://127.0.0.1:8000"):
        configs = {server_key: _cfg(url)}
        return ToolTransportInvoker(http=httpx.AsyncClient(), server_configs=configs)

    @pytest.mark.asyncio
    async def test_health_unavailable_returns_error_no_transport_call(self):
        invoker = self._make_invoker()
        registry = MagicMock()
        registry.is_unavailable.return_value = True
        registry.get_state.return_value = McpServerHealthState.OPEN
        invoker.set_health_registry(registry)
        mock_transport = AsyncMock()
        invoker._transports["srv"] = mock_transport
        result = await invoker.invoke("srv", "tool", {})
        assert result.is_error is True
        mock_transport.call.assert_not_called()

    @pytest.mark.asyncio
    async def test_half_open_allows_trial_dispatch(self):
        invoker = self._make_invoker()
        registry = MagicMock()
        registry.get_state.return_value = McpServerHealthState.HALF_OPEN
        registry.is_unavailable.return_value = False
        registry.record_success = MagicMock()
        invoker.set_health_registry(registry)
        mock_transport = AsyncMock(return_value=_ok())
        invoker._transports["srv"] = mock_transport
        result = await invoker.invoke("srv", "tool", {})
        assert result.is_error is False
        mock_transport.call.assert_called_once()

    @pytest.mark.asyncio
    async def test_lifecycle_ensure_ready_called(self):
        invoker = self._make_invoker()
        lifecycle = AsyncMock()
        invoker.set_lifecycle(lifecycle)
        mock_transport = AsyncMock(return_value=_ok())
        invoker._transports["srv"] = mock_transport
        await invoker.invoke("srv", "tool", {})
        lifecycle.ensure_ready.assert_awaited_once_with("srv")

    @pytest.mark.asyncio
    async def test_semaphore_limits_concurrency(self):
        configs = {"srv": _cfg()}
        invoker = ToolTransportInvoker(
            http=httpx.AsyncClient(),
            server_configs=configs,
            concurrency_limits={"srv": 1},
        )
        call_count = 0
        max_concurrent = 0
        inflight = 0
        async def slow_call(tool_name, args):
            nonlocal call_count, max_concurrent, inflight
            inflight += 1
            max_concurrent = max(max_concurrent, inflight)
            await asyncio.sleep(0.01)
            inflight -= 1
            call_count += 1
            return _ok()
        invoker._transports["srv"] = MagicMock(call=slow_call)
        await asyncio.gather(
            invoker.invoke("srv", "t", {}),
            invoker.invoke("srv", "t", {}),
        )
        assert max_concurrent == 1

    @pytest.mark.asyncio
    async def test_success_records_health_success(self):
        invoker = self._make_invoker()
        registry = MagicMock()
        registry.is_unavailable.return_value = False
        registry.get_state.return_value = MagicMock(value="healthy")
        invoker.set_health_registry(registry)
        invoker._transports["srv"] = AsyncMock(return_value=_ok())
        await invoker.invoke("srv", "tool", {})
        registry.record_success.assert_called_once_with("srv")

    @pytest.mark.asyncio
    async def test_tool_error_increments_stat_tool_errors(self):
        invoker = self._make_invoker()
        invoker._transports["srv"] = AsyncMock(return_value=_tool_err())
        await invoker.invoke("srv", "tool", {})
        assert invoker.stat_tool_errors.get("srv", 0) == 1

    @pytest.mark.asyncio
    async def test_transport_error_increments_stat_transport_errors(self):
        invoker = self._make_invoker()
        invoker._transports["srv"] = AsyncMock(side_effect=TransportError("conn refused"))
        await invoker.invoke("srv", "tool", {})
        assert invoker.stat_transport_errors.get("srv", 0) == 1

    @pytest.mark.asyncio
    async def test_transport_error_records_health_failure(self):
        invoker = self._make_invoker()
        registry = MagicMock()
        registry.is_unavailable.return_value = False
        registry.get_state.return_value = MagicMock(value="healthy")
        invoker.set_health_registry(registry)
        invoker._transports["srv"] = AsyncMock(side_effect=TransportError("err"))
        await invoker.invoke("srv", "tool", {})
        registry.record_failure.assert_called_once_with("srv")

    @pytest.mark.asyncio
    async def test_transport_error_returns_error_type_transport(self):
        invoker = self._make_invoker()
        invoker._transports["srv"] = AsyncMock(side_effect=TransportError("err"))
        result = await invoker.invoke("srv", "tool", {})
        assert result.is_error is True
        assert result.error_type == "transport"

    @pytest.mark.asyncio
    async def test_missing_transport_returns_tool_error(self):
        invoker = ToolTransportInvoker(http=httpx.AsyncClient(), server_configs={})
        result = await invoker.invoke("nonexistent_srv", "tool", {})
        assert result.is_error is True
        assert result.error_type == "tool"

    def test_unknown_concurrency_limit_key_logs_warning(self, caplog):
        import logging
        with caplog.at_level(logging.WARNING):
            ToolTransportInvoker(
                http=httpx.AsyncClient(),
                server_configs={"srv": _cfg()},
                concurrency_limits={"nonexistent_key": 5},
            )
        assert "nonexistent_key" in caplog.text or "unknown" in caplog.text.lower()
```
