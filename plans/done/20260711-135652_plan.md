## Goal

Make live-`/v1/tools` drift diagnostics distinguish "truly unknown tool" from "known tool owned by a different server," and make `tool_names`'s validation-only semantics (including the omitted/empty-skips-validation behavior) explicit and example-backed in documentation — without duplicating the cross-server duplicate-ownership detection that already exists and is already tested.

## Design

### `validate_live_tools_match()` message differentiation (tool_registry.py)

```python
def validate_live_tools_match(
    self,
    server_key: str,
    live_tool_names: list[str],
) -> list[str]:
    """Validate live /v1/tools response against registry. Returns list of mismatches.

    Distinguishes an unknown tool (not registered to any server) from a tool
    that IS registered, but to a different server than the one whose live
    response claimed it — the latter names the actual registry owner.
    """
    registry_tools = set(self.get_tool_names(server_key))
    live_set = set(live_tool_names)

    mismatches: list[str] = []
    in_live_not_registry = live_set - registry_tools
    for t in sorted(in_live_not_registry):
        owner = self.get_server_for_tool(t)
        if owner is None:
            mismatches.append(
                f"[{server_key}] tool {t!r} in live response is unknown "
                f"(not registered to any server)"
            )
        else:
            mismatches.append(
                f"[{server_key}] tool {t!r} in live response is registered to "
                f"{owner!r}, not {server_key!r}"
            )

    in_registry_not_live = registry_tools - live_set
    if in_registry_not_live:
        mismatches.extend(
            f"[{server_key}] tool {t!r} in registry but not in live response"
            for t in sorted(in_registry_not_live)
        )

    return mismatches
```

(The "registry but not in live" branch — Task "Preserve existing detection for tools in registry but missing from live response" — is unchanged.)

### Test strengthening (tests/test_tool_registry.py)

```python
def test_live_tool_unknown_message_is_explicit(self) -> None:
    registry = ToolRegistry()
    registry.register(ToolDefinition(name="tool_a", server_key="server_x"))
    drift = validate_routing_against_live(
        registry, {"server_x": ["tool_a", "totally_unknown_tool"]}
    )
    assert any(
        "unknown" in msg and "totally_unknown_tool" in msg
        for msg in drift["server_x"]
    )

def test_live_tool_wrong_owner_message_names_owner(self) -> None:
    registry = ToolRegistry()
    registry.register(ToolDefinition(name="tool_x", server_key="server_a"))
    drift = validate_routing_against_live(
        registry, {"server_a": [], "server_b": ["tool_x"]}
    )
    assert any(
        "'server_a'" in msg and "tool_x" in msg for msg in drift["server_b"]
    ), drift["server_b"]
```

The existing `test_live_returns_tool_under_wrong_server` and `test_live_returns_tool_not_in_registry` tests keep their current (weaker, tool-name-only) assertions — these two new tests add the message-*content* precision the requirement targets, rather than replacing already-passing coverage.
