## Design

### `rebuild_vec()` — rebuild `chunks_vec` from `chunks`
Canonical data: `chunks` table. Rebuild: delete all `chunks_vec` rows, re-insert from `chunks.embedding`.

```python
def rebuild_vec(self) -> int:
    """Rebuild chunks_vec from chunks. Returns number of rows re-inserted."""
    with SQLiteHelper(cfg=...) as db:
        deleted = db.execute("DELETE FROM chunks_vec").rowcount
        rows = db.execute("SELECT chunk_id, embedding FROM chunks WHERE embedding IS NOT NULL").fetchall()
        for chunk_id, emb in rows:
            db.execute("INSERT INTO chunks_vec(chunk_id, embedding) VALUES(?, ?)", (chunk_id, emb))
        return len(rows)
```

### `reconcile_url(url)` — reconcile a single URL
Delete all derived data for the URL and re-trigger ingestion from the canonical `chunks` record.

```python
def reconcile_url(self, url: str) -> dict:
    """Delete FTS and vec entries for url; returns counts deleted."""
    with SQLiteHelper(cfg=...) as db:
        doc = db.execute("SELECT doc_id FROM documents WHERE url = ?", (url,)).fetchone()
        if doc is None:
            return {"found": False}
        doc_id = doc["doc_id"]
        chunk_ids = [r["chunk_id"] for r in db.execute(
            "SELECT chunk_id FROM chunks WHERE doc_id = ?", (doc_id,)).fetchall()]
        # Delete derived tables
        for cid in chunk_ids:
            db.execute("DELETE FROM chunks_vec WHERE chunk_id = ?", (cid,))
        db.execute(f"DELETE FROM chunks_fts WHERE chunk_id IN ({','.join('?' * len(chunk_ids))})", chunk_ids)
        # Re-insert into FTS and vec from chunks
        for cid in chunk_ids:
            row = db.execute("SELECT * FROM chunks WHERE chunk_id = ?", (cid,)).fetchone()
            # FTS: trigger re-insert
            db.execute("INSERT INTO chunks_fts(chunk_id, content) VALUES(?, ?)", (cid, row["content"]))
            if row["embedding"]:
                db.execute("INSERT INTO chunks_vec(chunk_id, embedding) VALUES(?, ?)", (cid, row["embedding"]))
        return {"found": True, "chunks": len(chunk_ids)}
```
