## Design

### `tests/shared/test_tool_result_cache.py`

```python
import time
from unittest.mock import patch
import pytest
from shared.tool_cache import ToolResultCache
from shared.transport_dto import ToolCallResult

def _ok(output: str = "out") -> ToolCallResult:
    return ToolCallResult(output=output, is_error=False, request_id="req", server_key="srv")

def _err() -> ToolCallResult:
    return ToolCallResult(output="err", is_error=True, request_id="", server_key="", error_type="tool")

class TestToolResultCache:
    def test_make_key_includes_tool_name_and_args(self):
        c = ToolResultCache(ttl=60)
        k1 = c.make_key("tool_a", {"x": 1})
        k2 = c.make_key("tool_b", {"x": 1})
        assert k1 != k2
        assert "tool_a" in k1 or k1.startswith("tool_a:")

    def test_cache_hit_returns_result(self):
        c = ToolResultCache(ttl=60)
        key = c.make_key("t", {})
        c.store_if_success(key, _ok("hello"))
        result = c.get_result(key)
        assert result is not None
        assert result.output == "hello"
        assert result.request_id == ""
        assert result.server_key == ""

    def test_cache_miss_returns_none(self):
        c = ToolResultCache(ttl=60)
        assert c.get_result("nonexistent") is None

    def test_expired_entry_removed(self):
        c = ToolResultCache(ttl=60)
        key = c.make_key("t", {})
        c.store_if_success(key, _ok())
        with patch("shared.tool_cache.time") as mock_time:
            mock_time.time.return_value = time.time() + 120  # +120s past TTL
            assert c.get_result(key) is None

    def test_lru_move_on_hit(self):
        c = ToolResultCache(ttl=60, max_size=2)
        k1 = c.make_key("t1", {})
        k2 = c.make_key("t2", {})
        c.store_if_success(k1, _ok("1"))
        c.store_if_success(k2, _ok("2"))
        _ = c.get_result(k1)  # access k1 → moves to most-recently-used
        k3 = c.make_key("t3", {})
        c.store_if_success(k3, _ok("3"))  # evicts k2 (LRU), not k1
        assert c.get_result(k1) is not None
        assert c.get_result(k2) is None

    def test_max_size_evicts_oldest(self):
        c = ToolResultCache(ttl=60, max_size=2)
        k1 = c.make_key("t1", {})
        k2 = c.make_key("t2", {})
        k3 = c.make_key("t3", {})
        c.store_if_success(k1, _ok())
        c.store_if_success(k2, _ok())
        c.store_if_success(k3, _ok())  # evicts k1
        assert c.get_result(k1) is None
        assert c.get_result(k2) is not None
        assert c.get_result(k3) is not None

    def test_error_result_not_stored(self):
        c = ToolResultCache(ttl=60)
        key = c.make_key("t", {})
        c.store_if_success(key, _err())
        assert c.get_result(key) is None

    def test_clear_removes_all_entries(self):
        c = ToolResultCache(ttl=60)
        key = c.make_key("t", {})
        c.store_if_success(key, _ok())
        c.clear()
        assert c.get_result(key) is None

    def test_apply_config_updates_ttl(self):
        c = ToolResultCache(ttl=60, max_size=0)
        key = c.make_key("t", {})
        c.store_if_success(key, _ok())
        c.apply_config(ttl=0)  # set TTL to 0 → all entries immediately expired
        assert c.get_result(key) is None
```

### `tests/shared/test_tool_executor_stampede.py`

```python
import asyncio
from unittest.mock import AsyncMock, patch
import pytest
import httpx
from shared.tool_executor import ToolExecutor
from shared.transport_dto import ToolCallResult

def _make_executor() -> ToolExecutor:
    return ToolExecutor(http=httpx.AsyncClient(), cache_ttl=60.0, server_configs={})

def _ok_result() -> ToolCallResult:
    return ToolCallResult(output="ok", is_error=False, request_id="", server_key="srv")

def _err_result() -> ToolCallResult:
    return ToolCallResult(output="err", is_error=True, request_id="", server_key="srv", error_type="tool")

class TestStampedeProtection:
    @pytest.mark.asyncio
    async def test_concurrent_miss_calls_raw_execute_once(self):
        ex = _make_executor()
        call_count = 0
        async def mock_raw(tool_name, args):
            nonlocal call_count
            call_count += 1
            await asyncio.sleep(0)
            return _ok_result()
        with patch.object(ex, "_raw_execute", side_effect=mock_raw):
            r1, r2 = await asyncio.gather(
                ex._execute_with_cache("t", {}),
                ex._execute_with_cache("t", {}),
            )
        assert call_count == 1
        assert r1.output == r2.output == "ok"

    @pytest.mark.asyncio
    async def test_concurrent_success_same_result(self):
        ex = _make_executor()
        mock_result = _ok_result()
        with patch.object(ex, "_raw_execute", return_value=mock_result):
            r1, r2 = await asyncio.gather(
                ex._execute_with_cache("t", {}),
                ex._execute_with_cache("t", {}),
            )
        assert r1 is mock_result or r1.output == mock_result.output
        assert r2 is mock_result or r2.output == mock_result.output

    @pytest.mark.asyncio
    async def test_concurrent_exception_both_callers_raise(self):
        ex = _make_executor()
        async def mock_raise(tool_name, args):
            await asyncio.sleep(0)
            raise RuntimeError("transport failed")
        with patch.object(ex, "_raw_execute", side_effect=mock_raise):
            with pytest.raises(RuntimeError, match="transport failed"):
                await asyncio.gather(
                    ex._execute_with_stampede_protection("k", "t", {}),
                    ex._execute_with_stampede_protection("k", "t", {}),
                )

    @pytest.mark.asyncio
    async def test_inflight_cleaned_after_success(self):
        ex = _make_executor()
        with patch.object(ex, "_raw_execute", return_value=_ok_result()):
            await ex._execute_with_stampede_protection("key", "t", {})
        assert "key" not in ex._inflight

    @pytest.mark.asyncio
    async def test_inflight_cleaned_after_exception(self):
        ex = _make_executor()
        async def raise_error(*_):
            raise RuntimeError("err")
        with patch.object(ex, "_raw_execute", side_effect=raise_error):
            with pytest.raises(RuntimeError):
                await ex._execute_with_stampede_protection("key", "t", {})
        assert "key" not in ex._inflight

    @pytest.mark.asyncio
    async def test_failed_tool_result_not_cached(self):
        ex = _make_executor()
        with patch.object(ex, "_raw_execute", return_value=_err_result()):
            await ex._execute_with_cache("t", {})
        key = ex._cache.make_key("t", {})
        assert ex._cache.get_result(key) is None

    @pytest.mark.asyncio
    async def test_no_caller_hangs_on_exception(self):
        """All concurrent callers must receive exception within a short timeout."""
        ex = _make_executor()
        async def slow_raise(tool_name, args):
            await asyncio.sleep(0.01)
            raise RuntimeError("boom")
        with patch.object(ex, "_raw_execute", side_effect=slow_raise):
            with pytest.raises((RuntimeError, BaseException)):
                await asyncio.wait_for(
                    asyncio.gather(
                        ex._execute_with_stampede_protection("k", "t", {}),
                        ex._execute_with_stampede_protection("k", "t", {}),
                        return_exceptions=True,
                    ),
                    timeout=2.0,
                )
```

## Risks

| Risk | Likelihood | Mitigation |
|---|---|---|
| `test_expired_entry_removed` time-sensitive — may flake with real sleep | Medium | Patch `time.time` in `shared.tool_cache` instead of sleeping |
| `test_concurrent_exception_both_callers_raise` — `asyncio.gather` with `return_exceptions=False` (default) raises on first error; second caller exception is lost | Medium | Use `return_exceptions=True` and assert both results are `RuntimeError` instances; or catch the gathered exception and verify |
| `test_no_caller_hangs_on_exception` may be too slow on CI | Low | 2s timeout is generous; reduce to 1s or make the mock not sleep |
| `ToolResultCache` path not yet available until require-23 is implemented | High — dependency | Tests are written for require-23 outcome; run only after require-23 is implemented |
