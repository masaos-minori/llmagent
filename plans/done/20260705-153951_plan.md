**Static scan test:**
```python
import pathlib

SCRIPTS_DIR = pathlib.Path("scripts")
TEST_DIRS = {"tests", "test"}

def test_production_code_does_not_call_reset_registry():
    """CI check: _reset_registry_for_testing() must not appear in production code."""
    forbidden = "_reset_registry_for_testing"
    violations = []
    for py_file in SCRIPTS_DIR.rglob("*.py"):
        # Exclude test directories
        parts = set(py_file.parts)
        if parts & TEST_DIRS:
            continue
        content = py_file.read_text()
        if forbidden in content and "def _reset_registry_for_testing" not in content:
            violations.append(str(py_file))
    assert not violations, f"Production code calls {forbidden}: {violations}"

def test_allowed_usage_in_test_files():
    """Test files are allowed to import _reset_registry_for_testing."""
    from shared.tool_registry import _reset_registry_for_testing
    _reset_registry_for_testing()  # just verify it works
    from shared.tool_registry import get_registry
    registry = get_registry()
    assert registry is not None
```
