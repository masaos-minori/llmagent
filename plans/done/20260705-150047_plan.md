## Design

**Static scan test approach**:
```python
import ast, pathlib

AGENT_DIR = pathlib.Path("scripts/agent")
APPROVED_RAG_FILES = {"rag_maintenance_service.py"}
APPROVED_MDQ_FILES = set()  # none — no direct mdq access should exist

def find_db_calls(root: pathlib.Path, db_name: str) -> list[tuple[str, int]]:
    """Find SQLiteHelper("<db_name>") calls. Return (file, lineno) list."""
    hits = []
    for py_file in root.rglob("*.py"):
        src = py_file.read_text()
        if f'SQLiteHelper("{db_name}")' in src:
            hits.append((str(py_file.relative_to(root)), ...))
    return hits

def test_no_unauthorized_rag_db_access():
    hits = find_db_calls(AGENT_DIR, "rag")
    unauthorized = [f for f, _ in hits if f not in APPROVED_RAG_FILES]
    assert not unauthorized, f"Unauthorized rag DB access: {unauthorized}"

def test_no_unauthorized_mdq_db_access():
    hits = find_db_calls(AGENT_DIR, "mdq")
    assert not hits, f"Unauthorized mdq DB access: {hits}"
```
