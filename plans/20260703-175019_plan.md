## Design

### `tests/shared/test_plugin_tool_registration.py`

```python
import pytest
from pathlib import Path
from shared import plugin_registry

@pytest.fixture(autouse=True)
def reset():
    plugin_registry._reset_for_testing()
    yield
    plugin_registry._reset_for_testing()

class TestPluginToolRegistration:
    def test_valid_annotation_registers_successfully(self):
        """tuple[str, bool] annotation succeeds."""
        @plugin_registry.register_tool("valid_tool")
        async def handler(args: dict) -> tuple[str, bool]:
            return "ok", False
        assert plugin_registry.get_tool("valid_tool") is not None

    def test_missing_annotation_raises_value_error(self):
        """Missing return annotation raises ValueError at decoration time."""
        with pytest.raises(ValueError, match="missing return type annotation"):
            @plugin_registry.register_tool("bad_tool")
            async def handler(args):
                return "ok", False

    def test_wrong_annotation_raises_value_error(self):
        """Wrong annotation type raises ValueError."""
        with pytest.raises(ValueError, match="expected return type"):
            @plugin_registry.register_tool("wrong_tool")
            async def handler(args: dict) -> str:
                return "x"

    def test_invalid_plugin_not_in_registry(self):
        """After a failed registration, the tool name is absent from the registry."""
        try:
            @plugin_registry.register_tool("absent_tool")
            async def handler(args):
                return "ok", False
        except ValueError:
            pass
        assert plugin_registry.get_tool("absent_tool") is None

    def test_non_strict_load_records_failure_and_continues(self, tmp_path: Path):
        """Non-strict load: annotation failure recorded as PluginFailure, loading continues."""
        (tmp_path / "bad.py").write_text(
            "from shared.plugin_registry import register_tool\n"
            "@register_tool('bad')\n"
            "async def h(args): return 'x', False\n"
        )
        (tmp_path / "ok.py").write_text(
            "from shared.plugin_registry import register_tool\n"
            "@register_tool('ok')\n"
            "async def h(args: dict) -> tuple[str, bool]: return 'x', False\n"
        )
        result = plugin_registry.load_plugins(tmp_path)
        assert result.loaded_count == 1
        assert len(result.failed) == 1
        assert "missing return type annotation" in result.failed[0].error

    def test_strict_mode_raises_plugin_load_error(self, tmp_path: Path):
        """Strict mode: annotation failure is aggregated into PluginLoadError."""
        (tmp_path / "bad.py").write_text(
            "from shared.plugin_registry import register_tool\n"
            "@register_tool('bad')\n"
            "async def h(args): return 'x', False\n"
        )
        with pytest.raises(plugin_registry.PluginLoadError):
            plugin_registry.load_plugins(tmp_path, strict_mode=True)
```

### `tests/shared/test_plugin_tool_invoker.py`

```python
import pytest
from shared import plugin_registry
from shared.plugin_tool_invoker import PluginToolInvoker
from shared.transport_dto import ToolCallResult

@pytest.fixture(autouse=True)
def reset():
    plugin_registry._reset_for_testing()
    yield
    plugin_registry._reset_for_testing()

@pytest.fixture
def invoker():
    return PluginToolInvoker()

class TestPluginToolInvoker:
    @pytest.mark.asyncio
    async def test_missing_plugin_returns_none(self, invoker):
        result = await invoker.try_execute("nonexistent", {})
        assert result is None

    @pytest.mark.asyncio
    async def test_valid_plugin_returns_tool_call_result(self, invoker):
        async def fn(args: dict) -> tuple[str, bool]:
            return "output", False
        plugin_registry._tools["tool_a"] = (fn, "test")
        result = await invoker.try_execute("tool_a", {})
        assert isinstance(result, ToolCallResult)
        assert result.output == "output"
        assert result.is_error is False

    @pytest.mark.asyncio
    async def test_plugin_exception_returns_error_result(self, invoker):
        async def fn(args: dict) -> tuple[str, bool]:
            raise RuntimeError("boom")
        plugin_registry._tools["bad_tool"] = (fn, "test")
        result = await invoker.try_execute("bad_tool", {})
        assert result is not None
        assert result.is_error is True
        assert result.error_type == "tool"

    @pytest.mark.asyncio
    async def test_result_uses_empty_request_id(self, invoker):
        async def fn(args: dict) -> tuple[str, bool]:
            return "x", False
        plugin_registry._tools["req_id_tool"] = (fn, "test")
        result = await invoker.try_execute("req_id_tool", {})
        assert result is not None
        assert result.request_id == ""

    @pytest.mark.asyncio
    async def test_result_uses_empty_server_key(self, invoker):
        async def fn(args: dict) -> tuple[str, bool]:
            return "x", False
        plugin_registry._tools["sk_tool"] = (fn, "test")
        result = await invoker.try_execute("sk_tool", {})
        assert result is not None
        assert result.server_key == ""

    @pytest.mark.asyncio
    async def test_invalid_tuple_length_raises(self, invoker):
        async def fn(args: dict):
            return ("only_one_element",)
        plugin_registry._tools["bad_len"] = (fn, "test")
        with pytest.raises(ValueError, match="must return exactly"):
            await invoker.try_execute("bad_len", {})

    @pytest.mark.asyncio
    async def test_wrong_output_type_raises(self, invoker):
        async def fn(args: dict):
            return (123, False)
        plugin_registry._tools["bad_type"] = (fn, "test")
        with pytest.raises(TypeError, match="output must be str"):
            await invoker.try_execute("bad_type", {})
```
