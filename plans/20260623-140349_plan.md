## Design

### Change 1: `cmd_config_display.py` — `_print_memory_settings()`

Add after `_print_tool_safety_settings()`:

```python
def _print_memory_settings(self, ctx: AgentContext) -> None:
    self._out.write("Memory layer settings:")
    self._out.write(f"  use_memory_layer    : {ctx.cfg.memory.use_memory_layer}")
    self._out.write(f"  memory_embed_enabled: {ctx.cfg.memory.memory_embed_enabled}")
    self._out.write(f"  memory_jsonl_dir    : {ctx.cfg.memory.memory_jsonl_dir or '(not set)'}")
    self._out.write(
        f"  max_inject_semantic : {ctx.cfg.memory.memory_max_inject_semantic}"
    )
    self._out.write(
        f"  max_inject_episodic : {ctx.cfg.memory.memory_max_inject_episodic}"
    )
    self._out.write(
        f"  min_importance      : {ctx.cfg.memory.memory_min_importance}"
    )
```


### Change 3: `tests/test_agent_cmd_config.py` — Memory settings test

Add to `TestPrintConfigValues`:

```python
class TestPrintMemorySettings:
    def test_memory_settings_shown_in_config(self, capsys: Any) -> None:
        """_print_config_values() must include a memory layer section."""
        cmd = _make_config_cmd()
        cmd._print_config_values()
        out = capsys.readouterr().out
        assert "use_memory_layer" in out
        assert "memory_embed_enabled" in out

    def test_memory_disabled_shows_correct_value(self, capsys: Any) -> None:
        """When use_memory_layer=False, output says False."""
        cmd = _make_config_cmd(use_memory_layer=False)
        cmd._print_config_values()
        out = capsys.readouterr().out
        assert "use_memory_layer    : False" in out

    def test_memory_enabled_shows_correct_value(self, capsys: Any) -> None:
        """When use_memory_layer=True, output says True."""
        cmd = _make_config_cmd(use_memory_layer=True)
        cmd._print_config_values()
        out = capsys.readouterr().out
        assert "use_memory_layer    : True" in out
```

Note: Review `_make_config_cmd()` factory in existing test to understand how to pass `use_memory_layer` — it may need a `memory` config sub-object added.
