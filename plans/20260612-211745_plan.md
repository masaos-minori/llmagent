### Step 1: ActionResult (`shared/action_result.py`)

```python
"""shared/action_result.py
Universal action/result contract for all machine-interpreted outputs.
"""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Literal

ActionType = Literal[
    "continue",
    "call_tool",
    "retrieve_more_context",
    "ask_user",
    "fail",
    "retry",
]

@dataclass(frozen=True)
class ActionResult:
    """Schema-driven result for any machine decision path."""
    action: ActionType
    reason: str = ""
    required_context: list[str] = field(default_factory=list)
    payload: dict[str, object] = field(default_factory=dict)
    errors: list[str] = field(default_factory=list)
    confidence: float = 1.0
```

### Step 2: TurnResult → frozen dataclass (`agent/orchestrator.py`)

**現状 (通常クラス):**
```python
class TurnResult:
    def __init__(self, success: bool, answer: str = "", error_kind: str | None = None):
        self.success = success
        self.answer = answer
        self.error_kind = error_kind
```

**変更後 (frozen dataclass):**
```python
from dataclasses import dataclass
from typing import Literal

@dataclass(frozen=True)
class TurnResult:
    """Typed result of one LLM turn."""
    action: Literal["continue", "fail"]
    answer: str = ""
    error_kind: str | None = None
    reason: str = ""

    @property
    def success(self) -> bool:
        """Backward-compatible: True when action == 'continue'."""
        return self.action == "continue"
```

**呼び出し箇所の変更 (orchestrator.py:147, 152):**
```python

return TurnResult(success=False, answer="", error_kind=str(e))

# After
return TurnResult(action="continue", answer=answer)
return TurnResult(action="fail", answer="", error_kind=str(e))
```

### Step 3: LLMTurnRunner.run() を TurnResult 返却に変更 (`agent/llm_turn_runner.py`)

**`run()` の変更:**
```python
# Before: -> str

    ...
    return await self._finalize_answer(message)  # str
    ...
    return "Maximum tool turns reached."

# After: -> TurnResult

    ...

    ...
```

**`_handle_llm_error()` の変更:**
```python
# Before: -> str
async def _handle_llm_error(self, e: LLMTransportError, turn: int) -> str:
    from agent.error_injection_service import ErrorInjectionService
    error_service = ErrorInjectionService(self._ctx)
    return error_service.inject_mid_turn_error(e, turn)

# After: -> TurnResult
async def _handle_llm_error(self, e: LLMTransportError, turn: int) -> "TurnResult":
    from agent.error_injection_service import ErrorInjectionService
    from agent.orchestrator import TurnResult
    error_service = ErrorInjectionService(self._ctx)
    summary = error_service.inject_mid_turn_error(e, turn)
    return TurnResult(action="fail", answer=summary, reason="llm_transport_error")
```

**`_handle_llm_turn()` の wrap 削除 (orchestrator.py:135-152):**
```python
# Before
answer = await self._llm_runner.run(llm_url)
...
return TurnResult(success=True, answer=answer)
# error case:
return TurnResult(success=False, answer="", error_kind=str(e))

# After: run() が TurnResult を返すので wrap 不要
result = await self._llm_runner.run(llm_url)
ctx.session.save("assistant", result.answer)
...
return result
# error case: run() 内で処理されるため orchestrator での catch 不要になる
```

> **注意:** `LLMTransportError` の catch が `_handle_llm_turn()` にあるが、`run()` 内部で `_handle_llm_error()` を呼んでいるため、orchestrator 側の `except LLMTransportError` は削除できる。ただし `on_error` コールバックの呼び出しは `_handle_llm_turn()` に残す必要があるため、`run()` がエラー時に `TurnResult(action="fail")` を返した場合は orchestrator で detect して `on_error` を呼ぶ形にする。

```python
"""agent/shared/health_models.py"""

from dataclasses import dataclass, field

@dataclass(frozen=True)
class ServiceWarning:
    label: str
    url: str
    message: str

@dataclass(frozen=True)
class HealthCheckResult:
    warnings: list[ServiceWarning] = field(default_factory=list)
```

### Step 7: `_parse_mqe_response()` 内部 DTO 化 (`rag/llm.py`)

**スコープ限定の理由:** `expand_queries()` の戻り値を変えると `rag/stages/mqe.py:_run_mqe()` → `PipelineContext.queries: list[str]` → `pipeline.py:search_queries()` まで波及する。今回は内部実装の型安全化のみ。

```python
# rag/llm.py に MqeParseResult を定義
@dataclass(frozen=True)
class MqeParseResult:
    """Internal typed result from MQE JSON parsing."""
    queries: list[str]  # original_query included as first element

def _parse_mqe_response(raw: str, original_query: str) -> MqeParseResult:
    """Parse and validate MQE JSON array; raises MqeParseError on failure."""
    m = re.search(r"\[.*\]", raw, re.DOTALL)
    if not m:
        raise MqeParseError(f"MQE response contains no JSON array: {raw!r}")
    try:
        expanded = orjson.loads(m.group())
    except orjson.JSONDecodeError as e:
        raise MqeParseError(f"MQE response JSON is malformed: {e}") from e
    if not isinstance(expanded, list):
        raise MqeParseError(f"MQE response JSON is not a list: {type(expanded).__name__}")
    valid = [q for q in expanded if isinstance(q, str) and q.strip()]
    logger.info(f"MQE: {len(valid)} queries expanded from original")
    return MqeParseResult(queries=[original_query] + valid)

# expand_queries() は list[str] のまま
async def expand_queries(self, query: str, context: str = "") -> list[str]:
    ...
    result = _parse_mqe_response(raw, query)
    return result.queries  # list[str] を返して外部 API を変えない
```

### Risk 2 — `on_error` コールバックと `_handle_llm_transport_error()` が実質デッドコード化している (critical)

**現状のバグ:**

`orchestrator.py:_handle_llm_turn()` には `except LLMTransportError` があるが、`llm_turn_runner.py:run()` が `LLMTransportError` を内部でキャッチして `str` を返すため、この `except` は**実際には到達しない（デッドコード）**。

```
run() が LLMTransportError をキャッチ → _handle_llm_error() が合成メッセージを history に注入 → str を返す
→ orchestrator の except LLMTransportError には到達しない
→ _handle_llm_transport_error() と on_error が呼ばれない ← バグ
```

テスト `test_on_error_called_for_partial_completion` (line 131) は `patch.object(orch._llm_runner, "run", AsyncMock(side_effect=err))` で `run()` 自体を raise するようにモックしているため通るが、本物の実行では `on_error` は呼ばれない。

**Step 3 での解決策: `TurnResult` に `exception` フィールドを追加する**

```python
# orchestrator.py の TurnResult に exception フィールドを追加
from __future__ import annotations
import dataclasses
from typing import TYPE_CHECKING, Literal
if TYPE_CHECKING:
    from shared.llm_client import LLMTransportError

@dataclasses.dataclass(frozen=True)
class TurnResult:
    action: Literal["continue", "fail"]
    answer: str = ""
    error_kind: str | None = None
    reason: str = ""
    exception: "LLMTransportError | None" = None  # ← 追加

    @property
    def success(self) -> bool:
        return self.action == "continue"
```

`llm_turn_runner._handle_llm_error()` を変更:
```python
# Before: str を返す
async def _handle_llm_error(self, e: LLMTransportError, turn: int) -> str:
    ...
    return err.summary

# After: TurnResult を返す
async def _handle_llm_error(self, e: LLMTransportError, turn: int) -> "TurnResult":
    from agent.orchestrator import TurnResult
    ...
    return TurnResult(action="fail", answer=err.summary, reason="llm_transport_error", exception=e)
```

`orchestrator._handle_llm_turn()` に `result.exception` の検出を追加:
```python
async def _handle_llm_turn(self, llm_url: str) -> TurnResult:
    ctx = self._ctx
    try:
        if self._on_turn_start:
            self._on_turn_start()
        with self._llm_runner._span_ctx("llm") as llm_span:
            llm_span.set_attribute("model_url", llm_url)
            result = await self._llm_runner.run(llm_url)
            logger.info(f"LLM response: {result.answer}")
            ctx.session.save("assistant", result.answer)
            # exception フィールドがあれば on_error と partial処理を呼ぶ
            if result.exception is not None:
                self._handle_llm_transport_error(result.exception, ctx)
                if self._on_error:
                    self._on_error(result.exception)
            else:
                if self._on_turn_end:
                    self._on_turn_end()
            return result
    except LLMTransportError as e:
        # run() が side_effect で直接 raise する場合 (テストモック or 将来の変更) に到達
        self._handle_llm_transport_error(e, ctx)
        if self._on_error:
            self._on_error(e)
        return TurnResult(action="fail", answer="", error_kind=str(e), exception=e)
```

**テストへの影響:**

- `test_on_error_called_for_partial_completion` (line 131): `run` を `side_effect=err` でモック → `except LLMTransportError` に到達 → `on_error` が呼ばれる → **テストはそのまま通る**
- `test_delegates_to_error_injection_service` (test_llm_turn_runner.py:89): `_handle_llm_error()` の戻り値が `str` → `TurnResult` に変わる → `assert result == expected` が `assert result.answer == expected` になる (**修正必要**)
