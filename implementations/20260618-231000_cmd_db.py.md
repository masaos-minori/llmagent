### Step 2: Refactor `_cmd_urls()` to use MCP tool call
Current (lines 73-84):
```python
async def _cmd_urls(self, ctx: CommandContext, args: list[str], **kwargs: Any) -> None:
    """List registered documents."""
    lang = kwargs.get("lang") or None
    limit = int(kwargs["limit"]) if "limit" in kwargs else 20
    docs = self._service.list_documents(lang, limit)
    if not docs:
        await ctx.send("(No documents found.)")
        return
    lines = [f"{d['url']} ({d['lang']}, {d['chunk_count']} chunks)" for d in docs]
    await ctx.send("\n".join(lines))
```

Change to:
```python
async def _cmd_urls(self, ctx: CommandContext, args: list[str], **kwargs: Any) -> None:
    """List registered documents via rag-pipeline-mcp."""
    params: dict[str, str | int] = {"limit": 20}
    if kwargs.get("lang"):
        params["lang"] = kwargs["lang"]
    result = await ctx.tool_executor.execute("rag_list_documents", params)
    if not result or result.strip() == "(No documents found.)":
        await ctx.send("(No documents found.)")
        return
    await ctx.send(result.strip())
```

### Step 3: Refactor `_cmd_clean()` to use MCP tool call
Current (lines 86-100):
```python
async def _cmd_clean(self, ctx: CommandContext, args: list[str], **kwargs: Any) -> None:
    """Delete a document and its chunks from DB by URL."""
    if not args:
        await ctx.send("Usage: /db clean <url>")
        return
    url = args[0]
    deleted = ctx.session.delete_document(url)
    if deleted:
        stats = self._service.stats()
        await ctx.send(f"Document deleted: {url}\nDocuments: {stats.docs}, Chunks: {stats.chunks}")
    else:
        await ctx.send(f"Document not found: {url}")
```

Change to:
```python
async def _cmd_clean(self, ctx: CommandContext, args: list[str], **kwargs: Any) -> None:
    """Delete a document via rag-pipeline-mcp."""
    if not args:
        await ctx.send("Usage: /db clean <url>")
        return
    url = args[0]
    result = await ctx.tool_executor.execute("rag_delete_document", {"url": url})
    if result and "not found" in result.lower():
        await ctx.send(result.strip())
    else:
        stats = self._service.stats()
        await ctx.send(f"{result.strip()}\nDocuments: {stats.docs}, Chunks: {stats.chunks}")
```
