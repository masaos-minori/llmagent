# Implementation Plan: Workflow Engine (Step 6)

## Goal

Implement `workflow_engine.py` with stage transitions, timeout enforcement, and retry logic based on `WorkflowDef` and `StateStore`.

## Scope

**In:**
- `scripts/agent/workflow/workflow_engine.py`: Stage execution loop, timeout, retry, halt logic

**Out:**
- Actual stage implementation (plan/execute/verify/retry are callbacks provided by caller)

## Assumptions

1. Stage implementations are callback functions passed to `WorkflowEngine.run()`.
2. Timeout uses `asyncio.wait_for()` for async stages or `signal.alarm()` for sync stages.
3. Retry uses exponential backoff: `delay = base_delay * (backoff_multiplier ** attempt_number)`.
4. `StateStore` is injected via constructor (not singleton).

```python
#!/usr/bin/env python3
"""workflow/workflow_engine.py — Stage transitions, timeout, retry control."""

from __future__ import annotations

import asyncio
import logging
import time
from dataclasses import dataclass, field
from typing import Any, Callable

from agent.workflow.models import (
    AttemptRecord,
    AttemptStatus,
    TaskRecord,
    TaskStatus,
    WorkflowDef,
    WorkflowStage,
)
from agent.workflow.state_store import StateStore

logger = logging.getLogger(__name__)

# Type alias for stage handler: async function receiving task_id and context
StageHandler = Callable[[str, dict[str, Any]], Any]


@dataclass
class ExecutionResult:
    """Result of a workflow execution."""
    task_id: str
    status: TaskStatus
    completed_stages: list[str] = field(default_factory=list)
    failed_stage: str | None = None
    error_message: str | None = None
    total_attempts: int = 0


class WorkflowEngine:
    """Executes workflow stages with timeout and retry support."""

    def __init__(self, workflow_def: WorkflowDef, state_store: StateStore) -> None:
        self._workflow_def = workflow_def
        self._store = state_store
        self._start_time: float = 0.0

    async def run(
        self,
        task: TaskRecord,
        handlers: dict[str, StageHandler],
        context: dict[str, Any] | None = None,
    ) -> ExecutionResult:
        """Execute all stages in workflow order.

        Args:
            task: TaskRecord to execute.
            handlers: Mapping of stage_id -> async handler function.
            context: Arbitrary context passed to each stage handler.

        Returns:
            ExecutionResult with final status and stage history.
        """
        self._start_time = time.monotonic()
        self._store.create_task(task)
        self._store.update_task_status(task.task_id, TaskStatus.RUNNING.value)

        result = ExecutionResult(task_id=task.task_id, status=TaskStatus.RUNNING)
        stage_order = self._workflow_def.stage_order()
        context = context or {}

        for stage_id in stage_order:
            # Check global timeout
            elapsed = time.monotonic() - self._start_time
            if elapsed > self._workflow_def.global_timeout_sec:
                logger.warning("Global timeout reached after %.1fs", elapsed)
                result.status = TaskStatus.HALTED
                result.failed_stage = stage_id
                result.error_message = f"Global timeout exceeded ({self._workflow_def.global_timeout_sec}s)"
                self._store.update_task_status(task.task_id, TaskStatus.HALTED.value)
                break

            # Check idempotency
            event_id = f"{task.task_id}:{stage_id}"
            idem_check = self._store.check_idempotency(event_id)
            if idem_check.is_duplicate:
                logger.info("Skipping duplicate event %s (artifact: %s)", event_id, idem_check.artifact_ref)
                result.completed_stages.append(stage_id)
                continue

            # Execute stage with retry
            stage_result = await self._execute_stage(
                task=task,
                stage_id=stage_id,
                handler=handlers.get(stage_id),
                context={**context, "event_id": event_id},
            )

            if stage_result == "completed":
                result.completed_stages.append(stage_id)
            else:
                result.status = TaskStatus.FAILED
                result.failed_stage = stage_id
                result.error_message = f"Stage {stage_id} failed after retries"
                self._store.update_task_status(task.task_id, TaskStatus.FAILED.value)
                break

        # Mark final status
        if result.status == TaskStatus.RUNNING:
            result.status = TaskStatus.COMPLETED
            self._store.update_task_status(task.task_id, TaskStatus.COMPLETED.value)

        return result

    async def _execute_stage(
        self,
        task: TaskRecord,
        stage_id: str,
        handler: StageHandler | None,
        context: dict[str, Any],
    ) -> str:
        """Execute a single stage with timeout and retry logic.

        Returns "completed" on success, "failed" on exhaustion of retries.
        """
        if handler is None:
            logger.warning("No handler for stage %s, skipping", stage_id)
            return "completed"

        stage_def = self._workflow_def.get_stage(stage_id)
        if stage_def is None:
            logger.error("Stage definition not found: %s", stage_id)
            return "failed"

        max_attempts = stage_def.retry.max_attempts
        backoff_multiplier = stage_def.retry.backoff_multiplier
        timeout_sec = stage_def.timeout_sec

        for attempt_num in range(1, max_attempts + 1):
            event_id = f"{task.task_id}:{stage_id}:attempt{attempt_num}"
            attempt_record = AttemptRecord(
                task_id=task.task_id,
                stage=WorkflowStage(stage_id),
                attempt_number=attempt_num,
            )

            try:
                # Create attempt record
                attempt_id = self._store.create_attempt(attempt_record)

                # Execute with timeout
                start = time.monotonic()
                result = await asyncio.wait_for(
                    handler(task.task_id, context),
                    timeout=timeout_sec,
                )
                elapsed = time.monotonic() - start

                # Mark attempt completed
                self._store.complete_attempt(attempt_id, AttemptStatus.COMPLETED.value)
                logger.info(
                    "Stage %s completed (attempt %d/%d, %.1fs)",
                    stage_id, attempt_num, max_attempts, elapsed,
                )

                # Record idempotency event
                artifact_ref = context.get("event_id", event_id)
                self._store.record_event(event_id=task.task_id + ":" + stage_id, task_id=task.task_id, stage_id=stage_id, artifact_ref=artifact_ref)

                return "completed"

            except asyncio.TimeoutError:
                logger.warning(
                    "Stage %s timeout on attempt %d/%d (%.1fs > %ds)",
                    stage_id, attempt_num, max_attempts,
                    time.monotonic() - start if 'start' in dir() else 0,
                    timeout_sec,
                )
                self._store.complete_attempt(
                    self._get_last_attempt_id(task.task_id),
                    AttemptStatus.TIMEOUT.value,
                    error=f"Timeout after {timeout_sec}s",
                )

            except Exception as e:
                logger.warning(
                    "Stage %s failed on attempt %d/%d: %s",
                    stage_id, attempt_num, max_attempts, e,
                )
                self._store.complete_attempt(
                    self._get_last_attempt_id(task.task_id),
                    AttemptStatus.FAILED.value,
                    error=str(e),
                )

            # Retry with backoff (except on last attempt)
            if attempt_num < max_attempts:
                delay = timeout_sec * 0.1 * (backoff_multiplier ** (attempt_num - 1))
                logger.info("Retrying stage %s in %.1fs", stage_id, delay)
                await asyncio.sleep(delay)

        # All attempts exhausted
        return "failed"

    def _get_last_attempt_id(self, task_id: str) -> int:
        """Get the most recent attempt_id for a task (for error recording)."""
        attempts = self._store.get_task_attempts(task_id)
        if attempts:
            return attempts[-1].attempt_id or 0
        return 0
```

### Details

- `run()` orchestrates the full workflow: create task → loop stages → mark final status
- Idempotency check at stage level skips already-completed stages (useful for recovery)
- Timeout per stage uses `asyncio.wait_for()`; global timeout checked before each stage
- Retry backoff: `delay = timeout_sec * 0.1 * (backoff_multiplier ^ (attempt_num - 1))`
- Failed stages update task status to FAILED; completed workflow updates to COMPLETED
