**Updated `_db_consistency()`:**
```python
    def _db_consistency(self) -> None:
        """Run RAG search index synchronization check."""
        try:
            with SQLiteHelper("rag").open() as db:
                report = check_rag_consistency(db)
        except Exception as e:  # noqa: BLE001 â€” rag.sqlite may not exist
            self._out.write_error(f"RAG consistency check failed: {e} [RAG]")
            return
        consistent = is_consistent(report)
        issues = summarize_issues(report)
        self._out.write(
            f"  chunks: {report.chunks}  fts: {report.fts}  vec: {report.vec}"
            f"  fts_gap: {report.fts_gap}  orphan_vec: {report.orphan_vec_count}"
            f"  fts_orphan: {report.fts_orphan_count}"
        )
        if consistent:
            self._out.write_success("RAG consistency: OK (chunks/FTS/vec in sync) [RAG]")
        else:
            for issue in issues:
                self._out.write_error(f"Consistency issue: {issue} [RAG]")
```
