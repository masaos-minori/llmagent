   ```python
   def test_check_sandbox_backend_ok_when_firejail():
       config = build_shell_config(sandbox_backend="firejail")
       result = _check_sandbox_backend(config)
       assert result.status == "ok"
   
   def test_check_sandbox_backend_warns_when_not_firejail():
       config = build_shell_config(sandbox_backend="none")
       result = _check_sandbox_backend(config)
       assert result.status == "warning"

   ```
3. Add firejail availability tests:
   ```python
   def test_require_firejail_ok_when_binary_present(mocker):
       mocker.patch("shutil.which", return_value="/usr/bin/firejail")
       config = build_shell_config(sandbox_backend="firejail")
       result = _require_firejail_available(config)
       assert result.status == "ok"
   
   def test_require_firejail_error_when_binary_absent(mocker):
       mocker.patch("shutil.which", return_value=None)
       config = build_shell_config(sandbox_backend="firejail")
       result = _require_firejail_available(config)
       assert result.status == "error"
   
   def test_require_firejail_ok_when_backend_not_firejail():
       config = build_shell_config(sandbox_backend="none")
       result = _require_firejail_available(config)
       assert result.status == "ok"  # binary check skipped for non-firejail backend
   ```
