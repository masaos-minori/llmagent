# Implementation: scripts/agent/workflow/workflow_engine.py

## Goal

Implement `WorkflowEngine` — orchestrates plan → execute → verify → (retry) stage transitions, enforces per-stage timeouts, applies the retry policy, persists state to `workflow.sqlite` via `StateStore`, and enforces idempotency via `processed_events`.

## Scope

**In:**
- `scripts/agent/workflow/workflow_engine.py` — new file

**Out:**
- Any direct LLM calls (delegated to `orchestrator.py` callbacks)
- Any MCP server changes
- UI or CLI output

## Assumptions

1. `WorkflowEngine` receives stage callback coroutines (`plan_fn`, `execute_fn`, `verify_fn`) from `orchestrator.py`; it does not call any agent code directly.
2. Timeout is enforced using `asyncio.wait_for`.
3. Idempotency check + event record share a `begin_immediate` transaction.
4. `event_id` = `f"{task.task_id}:{stage_id}:{attempt_number}"`.
5. If the idempotency key already exists in `processed_events`, the engine skips stage execution and returns the previous artifact URI (if any).
6. The `retry` stage is not a callback; it is internal state — the engine loops back to `execute_fn` after incrementing the attempt count.
7. On halt (max_attempts exceeded), the engine sets `task.status = "halted"` and raises `WorkflowHaltError`.

## Implementation

### Target file

`scripts/agent/workflow/workflow_engine.py`

### Procedure

Create the file with `WorkflowHaltError`, `WorkflowTimeoutError`, and `WorkflowEngine`.

### Method

```python
#!/usr/bin/env python3
"""agent/workflow/workflow_engine.py
Stage transition engine: plan → execute → verify → (retry loop).

Callers pass async callbacks for each stage. WorkflowEngine handles
timeout enforcement, retry counting, idempotency, and state persistence.
"""
from __future__ import annotations
import asyncio
import logging
from collections.abc import Awaitable, Callable

from agent.workflow.models import TaskRecord, WorkflowDef
from agent.workflow.state_store import StateStore

logger = logging.getLogger(__name__)

StageCallback = Callable[[], Awaitable[str | None]]  # returns artifact URI or None


class WorkflowHaltError(Exception):
    """Raised when a task exceeds max_attempts and is halted."""


class WorkflowTimeoutError(Exception):
    """Raised when a stage exceeds its timeout_sec."""


class WorkflowEngine:
    """Runs a workflow definition against a set of stage callbacks."""

    def __init__(self, workflow_def: WorkflowDef, store: StateStore) -> None:
        self._wdef = workflow_def
        self._store = store

    async def run(
        self,
        task: TaskRecord,
        plan_fn: StageCallback,
        execute_fn: StageCallback,
        verify_fn: StageCallback,
    ) -> None:
        """Execute plan → execute → verify with retry on execute failure."""
        self._store.update_task_status(task.task_id, "running")
        try:
            await self._run_stage(task, "plan", plan_fn)
            await self._run_execute_with_retry(task, execute_fn)
            await self._run_stage(task, "verify", verify_fn)
        except (WorkflowHaltError, WorkflowTimeoutError):
            self._store.update_task_status(task.task_id, "halted")
            raise
        except Exception:
            self._store.update_task_status(task.task_id, "failed")
            raise
        self._store.update_task_status(task.task_id, "completed")

    async def _run_execute_with_retry(
        self, task: TaskRecord, execute_fn: StageCallback
    ) -> None:
        """Run execute stage; retry up to max_attempts on failure."""
        policy = self._wdef.retry_policy
        attempt = 0
        while True:
            attempt += 1
            try:
                await self._run_stage(task, "execute", execute_fn, attempt=attempt)
                return
            except WorkflowTimeoutError:
                raise
            except Exception as exc:
                if attempt >= policy.max_attempts:
                    logger.error(
                        "Task %s: execute halted after %d attempts: %s",
                        task.task_id, attempt, exc,
                    )
                    raise WorkflowHaltError(
                        f"execute stage halted after {attempt} attempts"
                    ) from exc
                wait = policy.backoff_sec
                logger.warning(
                    "Task %s: execute attempt %d failed, retrying in %ds: %s",
                    task.task_id, attempt, wait, exc,
                )
                await asyncio.sleep(wait)

    async def _run_stage(
        self,
        task: TaskRecord,
        stage_id: str,
        fn: StageCallback,
        attempt: int = 1,
    ) -> None:
        """Run a single stage with idempotency check and timeout enforcement."""
        stage_def = self._wdef.get_stage(stage_id)
        timeout = stage_def.timeout_sec if stage_def else 60
        event_id = f"{task.task_id}:{stage_id}:{attempt}"

        # Idempotency check
        with self._store._db.begin_immediate():
            if self._store.is_event_processed(event_id):
                logger.info("Stage %s skipped (already processed): %s", stage_id, event_id)
                return
            attempt_rec = self._store.start_attempt(task.task_id, stage_id)
            self._store.record_event(event_id, task.task_id, stage_id)

        # Execute with timeout
        try:
            artifact_uri = await asyncio.wait_for(fn(), timeout=timeout)
        except asyncio.TimeoutError as exc:
            self._store.finish_attempt(attempt_rec.attempt_id, "failed", f"timeout after {timeout}s")
            raise WorkflowTimeoutError(
                f"stage {stage_id!r} timed out after {timeout}s"
            ) from exc
        except Exception as exc:
            self._store.finish_attempt(attempt_rec.attempt_id, "failed", str(exc))
            raise

        self._store.finish_attempt(attempt_rec.attempt_id, "completed")
        if artifact_uri:
            self._store.record_artifact(task.task_id, stage_id, artifact_uri)
        logger.info("Stage %s completed: task=%s", stage_id, task.task_id)
```

### Details

- `begin_immediate()` wraps the idempotency check + `record_event` to make them atomic; `start_attempt` is inside the same transaction.
- `_run_execute_with_retry` implements the `retry` pseudo-stage as an internal loop — no separate stage callback is needed.
- The engine does not import from `orchestrator.py`, keeping the dependency direction clean.
- `WorkflowTimeoutError` is re-raised immediately (no retry on timeout).
