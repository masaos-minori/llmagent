from db.config import get_db_config  # or wherever embedding_dims is read


def _check_embedding_dimensions(ctx: AgentContext) -> None:
    """Verify embedding dimension consistency across config layers."""
    memory_dim = ctx.cfg.memory.memory_embed_dim
    db_cfg = get_db_config()
    db_dim = db_cfg.embedding_dims

    if memory_dim != db_dim:
        logger.error(
            "Embedding dimension mismatch: memory=%d, db=%d. "
            "Fix config/memory.toml or db/config.py.",
            memory_dim, db_dim,
        )
        raise RuntimeError(
            f"Embedding dimension mismatch: memory={memory_dim}, db={db_dim}"
        )

    logger.info("Embedding dimensions consistent: %d", memory_dim)


def _check_services(ctx: AgentContext) -> None:
    audit_security_defaults(ctx.cfg)
    _check_embedding_dimensions(ctx)  # NEW
    check_readiness(ctx)
    check_tool_definitions_runtime(ctx)
