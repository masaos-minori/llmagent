def save_diagnostic(self, content: str) -> None:
    """Persist a diagnostic-only message; not included in normal history retrieval."""
    self._message_repo.save(
        role=_DIAGNOSTIC_ROLE,
        content=content,
        _diagnostic=True,
    )
