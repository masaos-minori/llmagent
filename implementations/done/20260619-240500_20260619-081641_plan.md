### Add new method to DiagnosticStore:

def save_serialization_event(
    self,
    session_id: int | None,
    round_id: str,
    trigger_tool: str,
    affected_count: int,
    mode: str,
    elapsed_ms: float,
    reason: str,
) -> None:
    """Persist a round-level serialization event."""
    import json
    content = json.dumps({
        "round_id": round_id,
        "trigger_tool": trigger_tool,
        "affected_count": affected_count,
        "mode": mode,
        "elapsed_ms": round(elapsed_ms, 1),
        "reason": reason,
    })
    self.save(
        session_id=session_id,
        kind="serialization_event",
        content=content,
    )
```

from agent.tool_runner import get_serialization_stats
import json

stats = get_serialization_stats()
if stats["total_events"] > 0:
    # Fetch serialization events from DiagnosticStore for breakdown
    events = []
    if ctx.services.diagnostic_store and ctx.session_id:
        diag_entries = ctx.services.diagnostic_store.fetch(ctx.session_id)
        for entry in diag_entries:
            if entry.get("kind") == "serialization_event":
                try:
                    events.append(json.loads(entry.get("content", "{}")))
                except (json.JSONDecodeError, TypeError):
                    pass
    
    # Calculate breakdown
    total_events = len(events) or stats["total_events"]
    reason_counts: dict[str, int] = {}
    trigger_counts: dict[str, int] = {}
    total_affected = 0
    
    for e in events:
        reason = e.get("reason", "unknown")
        reason_counts[reason] = reason_counts.get(reason, 0) + 1
        trigger = e.get("trigger_tool", "unknown")
        trigger_counts[trigger] = trigger_counts.get(trigger, 0) + 1
        total_affected += e.get("affected_count", 0)
    
    avg_affected = total_affected / total_events if total_events > 0 else 0
    
    self._out.write(f"  Serialization: {total_events} events (avg {avg_affected:.1f} tools/event)")
    if reason_counts:
        reason_str = ", ".join(f"{k}: {v}" for k, v in sorted(reason_counts.items()))
        self._out.write(f"    By reason: {reason_str}")
    if trigger_counts:
        top_triggers = sorted(trigger_counts.items(), key=lambda x: -x[1])[:5]
        trigger_str = ", ".join(f"{k} ({v})" for k, v in top_triggers)
        self._out.write(f"    Top triggers: {trigger_str}")
```

**Completion criterion:** `/mcp stats` displays serialization metrics with breakdown.

## Step 6: Add timing comparison logic

**Status:** Not implemented

**Target file:** `scripts/agent/tool_runner.py`

### Timing estimation formula:

For each round with serialization, estimate parallel time as the sum of the fastest individual tool execution (conservative estimate):

```python
def _estimate_parallel_time(tool_timings: dict[str, float]) -> float:
    """Estimate parallel execution time as sum of fastest individual tool times."""
    if not tool_timings:
        return 0.0
    # Each tool runs once; parallel time = max of all tool times (true parallel)
    # But for conservative estimate, use sum (sequential lower bound)
    return sum(tool_timings.values())

def _compute_serial_overhead(actual_ms: float, estimated_parallel_ms: float) -> float:
    """Compute ratio of actual serial time to estimated parallel time."""
    if estimated_parallel_ms <= 0:
        return 1.0
    return round(actual_ms / estimated_parallel_ms, 2)
```

### Capture per-tool timing in _execute_standard():

```python

for tc in approved_calls:
    # After each tool executes, record its elapsed time

```

### Store overhead ratio in serialization event:

```python
estimated_parallel = _estimate_parallel_time(tool_timings)
serial_overhead = _compute_serial_overhead(actual_elapsed_ms, estimated_parallel)

round_event["estimated_parallel_ms"] = round(estimated_parallel, 1)
round_event["serial_overhead"] = serial_overhead
ctx.stats.stat_serialization_total_overhead_ms += (actual_elapsed_ms - estimated_parallel)
```
