```python
def _note_list(self) -> None:
    notes = self._ctx.session.list_notes()
    if not notes:
        self._out.write_no_data("No notes.")
        return
    rows = [
        [
            str(n["note_id"]),
            "✓" if n.get("pinned") else "",
            n["created_at"][:19],
            (
                n["content"][:NOTE_CONTENT_TRUNCATE_AT] + "..."
                if len(n["content"]) > NOTE_CONTENT_MAX_CHARS
                else n["content"]
            ),
        ]
        for n in notes
    ]
    self._out.write_table(["ID", "Pin", "Created", "Content"], rows)
```

**`cmd_notes.py` — 3 new handlers:**
```python
def _note_pin(self, arg: str) -> None:
    if not arg.isdigit():
        self._out.write_validation_error("/note pin <id>")
        return
    ok = self._ctx.session.pin_note(int(arg))
    if ok:
        self._out.write_success(f"Note {arg} pinned.")
    else:
        self._out.write_no_data(f"Note {arg} not found.")

def _note_unpin(self, arg: str) -> None:
    if not arg.isdigit():
        self._out.write_validation_error("/note unpin <id>")
        return
    ok = self._ctx.session.unpin_note(int(arg))
    if ok:
        self._out.write_success(f"Note {arg} unpinned.")
    else:
        self._out.write_no_data(f"Note {arg} not found.")

def _note_search(self, query: str) -> None:
    if not query:
        self._out.write_validation_error("/note search <query>")
        return
    notes = self._ctx.session.search_notes(query)
    if not notes:
        self._out.write_no_data("No matching notes.")
        return
    rows = [
        [
            str(n["note_id"]),
            "✓" if n.get("pinned") else "",
            n["created_at"][:19],
            (
                n["content"][:NOTE_CONTENT_TRUNCATE_AT] + "..."
                if len(n["content"]) > NOTE_CONTENT_MAX_CHARS
                else n["content"]
            ),
        ]
        for n in notes
    ]
    self._out.write_table(["ID", "Pin", "Created", "Content"], rows)
```

**`test_agent_cmd_notes.py` — _make_cmd() extension:**
```python
def _make_cmd(
    *,
    add_return: int | None = 1,
    list_return: list | None = None,
    delete_return: bool = True,
    pin_return: bool = True,
    unpin_return: bool = True,
    search_return: list | None = None,
) -> _NotesMixin:
    session = MagicMock()
    session.add_note.return_value = add_return
    session.list_notes.return_value = list_return or []
    session.delete_note.return_value = delete_return
    session.pin_note.return_value = pin_return
    session.unpin_note.return_value = unpin_return
    session.search_notes.return_value = search_return or []
    ctx = SimpleNamespace(session=session)
    cmd = object.__new__(_NotesMixin)
    cmd._ctx = ctx
    return cmd
```

```python
class TestNotePin:
    def test_pin_success(self, capsys: pytest.CaptureFixture) -> None:
        cmd = _make_cmd(pin_return=True)
        cmd._cmd_note("pin 1")
        assert "pinned" in capsys.readouterr().out.lower()
        cmd._ctx.session.pin_note.assert_called_once_with(1)

    def test_pin_not_found(self, capsys: pytest.CaptureFixture) -> None:
        cmd = _make_cmd(pin_return=False)
        cmd._cmd_note("pin 99")
        assert "not found" in capsys.readouterr().out.lower()

    def test_pin_invalid_id_shows_usage(self, capsys: pytest.CaptureFixture) -> None:
        cmd = _make_cmd()

        assert "usage" in capsys.readouterr().out.lower()
        cmd._ctx.session.pin_note.assert_not_called()
```
