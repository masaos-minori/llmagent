    doc_id: int,
    chunk_index: int,
    content: str,
    normalized: str | None,
    chunk_type: str = "",
    source_file: str = "",

        """
        INSERT INTO chunks
            (doc_id, chunk_index, content, normalized_content, chunk_type, source_file)
        VALUES (?, ?, ?, ?, ?, ?)
        """,
        (doc_id, chunk_index, content, normalized, chunk_type, source_file),

    assert cur.lastrowid is not None
