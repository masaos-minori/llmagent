# Implementation: JsonlStore Write Serialization and Malformed-Line Counter

## Goal

Make `JsonlMemoryStore.write()` safe under concurrent async callers by adding an `asyncio.Lock`, and add a skip counter to `read_all()` for observability.

## Scope

**In:**
- Add `asyncio.Lock` to serialize `write()` calls inside `JsonlMemoryStore`
- Add a `_malformed_count: int` counter incremented when `read_all()` skips a bad line
- Expose `malformed_count` as a read-only property for metrics/testing

**Out:**
- File-level locking across multiple processes (single-event-loop assumption is maintained)
- JSONL compaction or checksum (deferred to a future task)

## Assumptions

- `scripts/agent/memory/jsonl_store.py`: `JsonlMemoryStore` class with `write(entry: MemoryEntry)` and `read_all() -> list[MemoryEntry]`
- The asyncio event loop is single-threaded; only concurrent coroutines within the same event loop need serialization
- `asyncio.Lock` is lazily created after event loop start (same pattern as `StdioTransport`)

## Implementation

### `scripts/agent/memory/jsonl_store.py`

```python
import asyncio

class JsonlMemoryStore:
    def __init__(self, path: str | Path) -> None:
        self._path = Path(path)
        self._lock: asyncio.Lock | None = None   # lazy: created after event loop starts
        self._malformed_count: int = 0

    def _get_lock(self) -> asyncio.Lock:
        if self._lock is None:
            self._lock = asyncio.Lock()
        return self._lock

    @property
    def malformed_count(self) -> int:
        return self._malformed_count

    async def write(self, entry: MemoryEntry) -> None:
        """Append one entry to the JSONL file; serialized by asyncio.Lock."""
        async with self._get_lock():
            line = orjson.dumps(asdict(entry)).decode() + "\n"
            self._path.parent.mkdir(parents=True, exist_ok=True)
            with self._path.open("a", encoding="utf-8") as f:
                f.write(line)

    def read_all(self) -> list[MemoryEntry]:
        """Read all entries; skip and count malformed lines."""
        if not self._path.exists():
            return []
        entries: list[MemoryEntry] = []
        with self._path.open("r", encoding="utf-8") as f:
            for lineno, line in enumerate(f, 1):
                line = line.strip()
                if not line:
                    continue
                try:
                    d = orjson.loads(line)
                    entry = _entry_from_dict(d)
                    if entry is not None:
                        entries.append(entry)
                    else:
                        self._malformed_count += 1
                        logger.warning(
                            "Skipping malformed JSONL line %d (count=%d)",
                            lineno, self._malformed_count,
                        )
                except orjson.JSONDecodeError as e:
                    self._malformed_count += 1
                    logger.warning(
                        "JSON decode error at line %d (count=%d): %s",
                        lineno, self._malformed_count, e,
                    )
        return entries
```

```python
async def test_concurrent_writes_are_serialized():
    """Two concurrent write() calls produce two lines, not garbled output."""
    store = JsonlMemoryStore(tmp_path / "mem.jsonl")
    entry1 = make_test_entry(content="first")
    entry2 = make_test_entry(content="second")
    await asyncio.gather(store.write(entry1), store.write(entry2))
    entries = store.read_all()
    assert len(entries) == 2

def test_malformed_line_increments_counter():
    """Malformed JSONL line increments malformed_count and is skipped."""
    path = tmp_path / "mem.jsonl"
    path.write_text('{"bad": "data"}\n{"memory_id": "x", ...valid...}\n')
    store = JsonlMemoryStore(path)
    entries = store.read_all()
    assert store.malformed_count >= 1
    assert len(entries) == 1
```
