def call_rag_service(
    http: httpx.Client,
    rag_url: str,
    query: str,
    history_context: str | None = None,
    auth_token: str | None = None,
    set_fetch_result: Callable[[TwoStageFetchResult], None] | None = None,
) -> str | None:
    """Delegate to external RAG service for context augmentation.

    Request details:
        - Endpoint: ``{rag_url}/v1/search``
        - Body: ``{"query": query, "history_context": history_context or ""}``
        - Headers: ``{"X-RAG-Token": auth_token}`` if auth_token is set
        - Timeout: 10.0 seconds

    Return contract:

        ┌─────────────┬──────────────────────────────────────────────────┐
        │ Return value │ Condition                                       │
        ├─────────────┼──────────────────────────────────────────────────┤
        │ ``str``      │ HTTP 200 + response body contains                │
        │              │ ``"context"`` key with non-empty string value    │
        │              │ Example: ``{"context": "relevant passage..."}``   │
        ├─────────────┼──────────────────────────────────────────────────┤
        │ ``""``       │ HTTP 200 but response body has no ``"context"``  │
        │              │ key, or ``"context"`` is null/empty. This is a   │
        │              │ valid (empty) result, NOT a failure.             │
        │              │ Example: ``{"context": null}`` or ``{"hits": []}``│
        ├─────────────┼──────────────────────────────────────────────────┤
        │ ``None``     │ One of:                                           │
        │              │ - HTTP 4xx (client error, no retry)              │
        │              │ - HTTP 5xx with all retries exhausted            │
        │              │ - Transport error (connection refused, timeout)   │
        │              │ - JSON parse error on response body               │
        │              │ None triggers in-process fallback in the caller.  │
        └─────────────┴──────────────────────────────────────────────────┘

    Retry behavior:
        - 5xx errors: retry up to ``_MAX_ATTEMPTS`` times with exponential backoff
        - Transport errors (connection refused, timeout): same retry policy
        - 4xx errors: no retry (client-side issue)
        - JSON parse errors: no retry (malformed response)

    Side effects:
        If ``set_fetch_result`` is provided, stores a ``TwoStageFetchResult``
        with fetch stage status and any hits from the response.

    Args:
        http: An initialized httpx.Client (caller manages lifecycle).
        rag_url: Base URL of the RAG service (e.g. ``http://127.0.0.1:8081``).
        query: The user query string to search for.
        history_context: Optional conversation history context appended to query.
        auth_token: Optional auth token sent as ``X-RAG-Token`` header.
        set_fetch_result: Optional callback to store fetch result metadata.

    Returns:
        Augmented context string, empty string for valid empty results,
        or None to signal failure and trigger fallback.
    """
```

def _augment_http(
    self, query: str, history_context: str | None = None
) -> str | None:
    """Run HTTP-stage augmentation via external RAG service.

    This is a thin wrapper around ``call_rag_service()``. All HTTP-level
    details (endpoint, headers, timeout, retries) are documented there.

    Return semantics:
        - ``str`` (non-empty): HTTP 200 with valid context → returned as final result
        - ``""`` (empty string): HTTP 200 with no context → returned as final result
          (not treated as failure; no fallback triggered)
        - ``None``: HTTP error, transport failure, or parse error → triggers
          in-process fallback chain (semantic cache → search pipeline → raw chunks)

    Side effects:
        Stores ``TwoStageFetchResult`` on ``self.last_fetch_result`` via the
        callback passed to ``call_rag_service()``.

    See Also:
        call_rag_service: Full return contract, HTTP details, and retry behavior.
    """

## Step 3: Add caller behavior documentation in `augment()` method

**Status:** Not implemented

**Current docstring:**

Run full pipeline and return a context block; '' when disabled or no results.

Fallback chain (each step produces the final result unless it signals failure):
1. HTTP mode: ``_augment_http()`` returns str/"" on success, None on failure.
   None triggers in-process fallback below.
2. Semantic cache lookup (in-process only): returns cached string or continues.
3. Search pipeline (semantic + FTS5 + merge + rerank): produces reranked hits.
4. Refiner: ``_augment_refiner()`` returns refined text on success, None on empty

5. Raw chunks: ``_format_chunks()`` formats reranked hits as the final context block.
```

**Changes needed — enhance:**
```python
def augment(
    self,
    query: str,
    debug_fn: Callable[[str], None] | None = None,
    history_context: str | None = None,
) -> str:
    """Run full pipeline and return a context block.

    Return values:
        - ``str`` (non-empty): Augmented context from one of the pipeline stages
        - ``""`` (empty string): Pipeline disabled (``use_search=False``), no cache
          hit, no search results, or refiner produced empty output

    Identity vs truthiness:
        The HTTP stage uses ``is not None`` identity checks (not truthiness).
        This means ``""`` from HTTP is treated as a valid result, while only
        explicit ``None`` triggers fallback. This distinction prevents the
        in-process pipeline from overriding a deliberate empty response from
        the RAG service.

    Fallback chain (each step produces the final result unless it returns None):
        1. HTTP mode: ``_augment_http()`` → str/"" (final) or None (fallback)
        2. Semantic cache: cached string (final) or None (fallback)
        3. Search pipeline: reranked hits → refiner → raw chunks formatting
        4. Refiner: refined text (final) or None (fallback to raw)
        5. Raw chunks: formatted text (final, may be empty)

    Side effects:
        - Updates ``self.last_stage_results`` with per-stage status
        - Updates ``self.last_fetch_result`` when HTTP stage is used
        - May update semantic cache on successful augment

    Raises:
        RagPipelineError: If the underlying database connection fails.
    """
```
