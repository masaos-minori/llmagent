## Proposed Changes

### 1. Add `IngestUrlResult` dataclass

```python
from dataclasses import dataclass

@dataclass(frozen=True)
class IngestUrlResult:
    url: str
    n_success: int       # chunks inserted into DB
    n_failed: int        # chunks that failed embedding/insertion
    skipped: bool        # True if document already registered (force=False)
```

### 2. Change `ingest_url_group()` return type

```python
def ingest_url_group(
    self,
    db: SQLiteHelper,
    url: str,
    chunk_files: list[Path],
    force: bool,
) -> IngestUrlResult:
    if not chunk_files:
        logger.warning("No chunk files provided for URL: %s", url)
        return IngestUrlResult(url=url, n_success=0, n_failed=0, skipped=True)

    chunk_files = sorted(chunk_files, key=lambda p: p.stem)
    first_data = self._read_chunk_json(chunk_files[0])
    if first_data is None:
        return IngestUrlResult(url=url, n_success=0, n_failed=len(chunk_files), skipped=False)

    title = first_data.get("title", "")
    lang = first_data.get("lang", "en")
    etag = first_data.get("etag")
    last_modified = first_data.get("last_modified")
    chunking_strategy = first_data.get("chunking_strategy", "text")
    doc_id = self._get_or_create_document(
        db, url, title, lang, force, etag, last_modified, chunking_strategy,
    )
    if doc_id is None:
        logger.info("already registered, skipping: %s", url)
        self._move_to_registered(chunk_files)
        return IngestUrlResult(url=url, n_success=0, n_failed=0, skipped=True)

    db.commit()
    inserted = self._ingest_chunk_files(doc_id, chunk_files)
    total = len(chunk_files)
    failed = total - inserted
    logger.info("inserted %s/%s chunks: %s", inserted, total, url)
    self._move_to_registered(chunk_files)
    return IngestUrlResult(url=url, n_success=inserted, n_failed=failed, skipped=False)
```

### 3. Aggregate results in `_process_url_groups()` and `ingest_all()`

```python
def _process_url_groups(
    self, db: SQLiteHelper, url_groups: dict[str, list[Path]], force: bool,
) -> list[IngestUrlResult]:
    results = []
    for url, paths in url_groups.items():
        try:
            result = self.ingest_url_group(db, url, paths, force)
            results.append(result)
        except (OSError, RuntimeError, ValueError, sqlite3.OperationalError):
            logger.exception("ingest_url_group failed: %s", url)
            results.append(IngestUrlResult(url=url, n_success=0, n_failed=-1, skipped=False))
    return results

def ingest_all(self, force: bool = False) -> None:
    chunk_files = sorted(self._chunk_dir.glob("*.txt"))
    if not chunk_files:
        logger.info("No chunk files to process")
        return
    url_groups = self._group_chunks_by_url(chunk_files)
    with SQLiteHelper().open(write_mode=True) as db:
        results = self._process_url_groups(db, url_groups, force)

    total_success = sum(r.n_success for r in results if r.n_failed >= 0)
    total_failed = sum(r.n_failed for r in results if r.n_failed >= 0 and r.n_failed != -1)
    total_skipped = sum(1 for r in results if r.skipped)
    logger.info("=== done: %s URLs processed (%d success, %d failed, %d skipped) ===",
                len(results), total_success, total_failed, total_skipped)
```
