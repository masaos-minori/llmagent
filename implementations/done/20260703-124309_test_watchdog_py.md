## Goal

Create `tests/test_watchdog.py` with four test cases covering the restart gating logic in `_watchdog_check_http()` and the early-exit fix in `watchdog_loop()`.

## Scope

- In-Scope:
  - Create `/home/masaos/llmagent/tests/test_watchdog.py` as a new file
  - Four test cases:
    1. HTTP 503 + `restart_recommended=True` → `lifecycle.restart()` is called
    2. HTTP 503 + `restart_recommended=False` → `lifecycle.restart()` is NOT called
    3. HTTP 200 + `ready=False` in body (with `restart_recommended=False`) → `lifecycle.restart()` NOT called
    4. `mcp_watchdog_interval=0` → `watchdog_loop()` returns immediately without sleeping
- Out-of-Scope:
  - Tests for `probe_mcp_health()` (covered in `test_repl_health.py`)
  - Tests for `operator_action_required` log message (lower priority; covered by Step 9)

## Assumptions

1. `_watchdog_check_http` requires `Steps 4 and 5` to be complete before these tests can pass.
2. `watchdog_loop` requires `Step 6` to be complete before test case 4 can pass.
3. Tests use `AsyncMock` and `MagicMock` from `unittest.mock`; no external dependencies beyond `pytest` and `pytest-asyncio`.
4. `AgentContext` is mocked via `MagicMock()` — the test sets only the attributes accessed by `_watchdog_check_http` / `watchdog_loop`.
5. `McpServerConfig` mock only needs `url`, `startup_mode`, and `transport` attributes.
6. The `_probe_mcp_health_detail` function is patched at `agent.repl_health._probe_mcp_health_detail` so the network call is replaced.
7. `asyncio.sleep` must be patched in test case 4 to detect if it is called; alternatively the test simply asserts the coroutine completes without awaiting sleep.

```python
"""tests/test_watchdog.py

Unit tests for watchdog restart gating logic and early-exit fix.

Requires Steps 4 (probe helper), 5 (watchdog check), and 6 (loop fix) to be applied.
"""

from __future__ import annotations

from unittest.mock import AsyncMock, MagicMock, patch

import pytest

from agent.repl_health import _watchdog_check_http, watchdog_loop
from agent.shared.health_models import McpHealthProbeResult


def _make_ctx(*, watchdog_interval: float = 30.0) -> MagicMock:
    """Build a minimal AgentContext mock."""
    ctx = MagicMock()
    ctx.cfg.mcp.mcp_watchdog_interval = watchdog_interval
    ctx.cfg.mcp.mcp_watchdog_max_restarts = 3
    ctx.cfg.mcp.mcp_servers = {}
    ctx.services_required.http = AsyncMock()
    ctx.services_required.lifecycle = AsyncMock()
    ctx.services_required.lifecycle.restart = AsyncMock()
    ctx.services_required.health_registry = None
    ctx.services_required.lifecycle.shutdown_idle = AsyncMock()
    return ctx


def _make_srv_cfg(*, url: str = "http://localhost:8000", startup_mode: str = "subprocess") -> MagicMock:
    srv_cfg = MagicMock()
    srv_cfg.url = url
    srv_cfg.startup_mode = startup_mode
    srv_cfg.transport = "http"
    return srv_cfg


# ── _watchdog_check_http() ────────────────────────────────────────────────────


class TestWatchdogCheckHttp:
    @pytest.mark.asyncio
    async def test_restart_called_when_503_and_restart_recommended(self) -> None:
        """HTTP 503 + restart_recommended=True triggers lifecycle.restart()."""
        ctx = _make_ctx()
        srv_cfg = _make_srv_cfg()
        restart_counts: dict[str, int] = {}

        probe_result = McpHealthProbeResult(
            reachable=True,
            status_code=503,
            restart_recommended=True,
            operator_action_required=False,
            body={"status": "degraded", "ready": False, "restart_recommended": True},
        )
        with patch(
            "agent.repl_health._probe_mcp_health_detail",
            new=AsyncMock(return_value=probe_result),
        ):
            await _watchdog_check_http(ctx, "test-server", srv_cfg, restart_counts, max_restarts=3)

        ctx.services_required.lifecycle.restart.assert_called_once_with("test-server")

    @pytest.mark.asyncio
    async def test_restart_not_called_when_503_and_restart_not_recommended(self) -> None:
        """HTTP 503 + restart_recommended=False: lifecycle.restart() must NOT be called."""
        ctx = _make_ctx()
        srv_cfg = _make_srv_cfg()
        restart_counts: dict[str, int] = {}

        probe_result = McpHealthProbeResult(
            reachable=True,
            status_code=503,
            restart_recommended=False,
            operator_action_required=True,
            body={"status": "degraded", "ready": False, "restart_recommended": False,
                  "operator_action_required": True},
        )
        with patch(
            "agent.repl_health._probe_mcp_health_detail",
            new=AsyncMock(return_value=probe_result),
        ):
            await _watchdog_check_http(ctx, "test-server", srv_cfg, restart_counts, max_restarts=3)

        ctx.services_required.lifecycle.restart.assert_not_called()

    @pytest.mark.asyncio
    async def test_restart_not_called_when_200_and_ready_false_in_body(self) -> None:
        """HTTP 200 with ready=False in body and restart_recommended=False: no restart."""
        ctx = _make_ctx()
        srv_cfg = _make_srv_cfg()
        restart_counts: dict[str, int] = {}

        probe_result = McpHealthProbeResult(
            reachable=True,
            status_code=200,
            restart_recommended=False,
            operator_action_required=False,
            body={"status": "ok", "ready": False, "restart_recommended": False},
        )
        with patch(
            "agent.repl_health._probe_mcp_health_detail",
            new=AsyncMock(return_value=probe_result),
        ):
            await _watchdog_check_http(ctx, "test-server", srv_cfg, restart_counts, max_restarts=3)

        ctx.services_required.lifecycle.restart.assert_not_called()


# ── watchdog_loop() ───────────────────────────────────────────────────────────


class TestWatchdogLoop:
    @pytest.mark.asyncio
    async def test_disabled_interval_returns_immediately_without_sleeping(self) -> None:
        """watchdog_loop() with interval=0 returns without sleeping."""
        ctx = _make_ctx(watchdog_interval=0)

        with patch("agent.repl_health.asyncio.sleep", new=AsyncMock()) as mock_sleep:
            await watchdog_loop(ctx)

        mock_sleep.assert_not_called()
```

3. Verify the file is placed at `tests/test_watchdog.py` (alongside other test files).

### Method

- `AsyncMock(return_value=probe_result)` for `_probe_mcp_health_detail` replaces the network call cleanly.
- Patching `agent.repl_health.asyncio.sleep` (not `asyncio.sleep`) ensures the patch targets the name as used in the module after import.
- `_make_ctx()` and `_make_srv_cfg()` are local factory helpers to keep each test compact.
- `pytest.mark.asyncio` requires `pytest-asyncio` which is already used in `test_repl_health.py`.
- Test case 3 uses HTTP 200 (fully healthy from a status-code perspective) but `ready=False` in the body with `restart_recommended=False`; this exercises the path that `probe.status_code == HTTPStatus.OK` triggers the success branch and returns without restarting.

### Details

- Import path: `from agent.repl_health import _watchdog_check_http, watchdog_loop`
- Import path: `from agent.shared.health_models import McpHealthProbeResult`
- The `_make_ctx` factory creates `ctx.services_required.lifecycle.shutdown_idle = AsyncMock()` to prevent attribute errors when `watchdog_loop` runs the shutdown_idle call (for tests that exercise the loop body, not applicable here since we return early).
- Test case 4 patches `agent.repl_health.asyncio.sleep` rather than `asyncio.sleep` because `repl_health.py` imports `asyncio` at the top level and uses `asyncio.sleep` directly.
