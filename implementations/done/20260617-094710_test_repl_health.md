```python
class TestCheckReadiness:
    """Tests for check_readiness() production/dev mode behavior."""

    def _make_ctx_with_http(self, mock_http) -> AgentContext:
        ctx = AgentContext()
        ctx.cfg = build_agent_config({
            "llm_url": "http://127.0.0.1:8001/v1/chat",
            "embed_url": "http://127.0.0.1:8003/embedding",
            "tool_definitions": [],
        })
        ctx.services = MagicMock()
        ctx.services.http = mock_http
        return ctx

    @pytest.mark.asyncio
    async def test_production_all_healthy_no_raise(self) -> None:
        mock_http = AsyncMock()
        mock_http.get.return_value = MagicMock(status_code=200)
        ctx = self._make_ctx_with_http(mock_http)
        result = await check_readiness(ctx, production_mode=True)
        assert not result.has_issues

    @pytest.mark.asyncio
    async def test_production_llm_down_raises(self) -> None:
        mock_http = AsyncMock()
        mock_http.get.side_effect = httpx.ConnectError("connection refused")
        ctx = self._make_ctx_with_http(mock_http)
        with pytest.raises(RuntimeError, match="Startup readiness check failed"):
            await check_readiness(ctx, production_mode=True)

    @pytest.mark.asyncio
    async def test_dev_mode_service_down_warns_only(self) -> None:
        mock_http = AsyncMock()
        mock_http.get.side_effect = httpx.ConnectError("connection refused")
        ctx = self._make_ctx_with_http(mock_http)
        result = await check_readiness(ctx, production_mode=False)
        assert result.has_issues  # warnings present, no exception

    @pytest.mark.asyncio
    async def test_dev_mode_healthy_no_issues(self) -> None:
        mock_http = AsyncMock()
        mock_http.get.return_value = MagicMock(status_code=200)
        ctx = self._make_ctx_with_http(mock_http)
        result = await check_readiness(ctx, production_mode=False)
        assert not result.has_issues
```
