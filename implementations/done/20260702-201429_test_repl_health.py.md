```python
class TestCheckRoutingDriftLive:
    @pytest.mark.asyncio
    async def test_no_drift_returns_empty(self):
        """No drift between live and registry → empty list."""
        # Live tool lists match registry → no drift
        with patch("agent.repl_health._collect_server_tool_lists", new_callable=AsyncMock) as mock_collect:
            mock_collect.return_value = {"file_read": ["read_file"], "file_write": ["write_file"]}
            result = await check_routing_drift_live(AgentContext())
        assert result == []

    @pytest.mark.asyncio
    async def test_drift_detected_returns_warnings(self):
        """Extra tool in live response not in registry → warning returned."""
        with patch("agent.repl_health._collect_server_tool_lists", new_callable=AsyncMock) as mock_collect:
            # Extra tool "unknown_tool" not in registry for file_read server
            mock_collect.return_value = {"file_read": ["read_file", "unknown_tool"]}
            result = await check_routing_drift_live(AgentContext())
        assert len(result) >= 1
        assert "Live routing drift [file_read]" in result[0]

    @pytest.mark.asyncio
    async def test_all_servers_unreachable_returns_empty(self):
        """All servers unreachable → empty list (no live data to validate)."""
        with patch("agent.repl_health._collect_server_tool_lists", new_callable=AsyncMock) as mock_collect:
            mock_collect.return_value = {}
            result = await check_routing_drift_live(AgentContext())
        assert result == []

    @pytest.mark.asyncio
    async def test_exception_is_swallowed(self):
        """Exception in _collect_server_tool_lists → empty list (non-fatal)."""
        with patch("agent.repl_health._collect_server_tool_lists", new_callable=AsyncMock) as mock_collect:
            mock_collect.side_effect = RuntimeError("network error")
            result = await check_routing_drift_live(AgentContext())
        assert result == []
```

### Method

- Use `pytest.mark.asyncio` for async test methods.
- Patch `_collect_server_tool_lists` at the module level where it's defined (`agent.repl_health`).
- Mock return values to simulate each scenario (no drift, drift detected, all unreachable, exception).
- Assert on the warning message format to verify correctness.

### Details

- `test_no_drift_returns_empty`: Mock returns tool lists that match the registry → no drift.
- `test_drift_detected_returns_warnings`: Mock returns an extra tool not in registry → drift warning with server_key prefix.
- `test_all_servers_unreachable_returns_empty`: Mock returns empty dict → skipped check, empty result.
- `test_exception_is_swallowed`: Mock raises exception → caught and swallowed, empty result.
