### Step 2: Implement methods in service.py

Add to `RagPipelineMCPService`:

```python
async def list_documents(self, lang: str | None = None, limit: int = 20) -> list[dict]:
    """List documents from rag.sqlite."""
    sql = (
        "SELECT d.url, d.title, d.lang, d.fetched_at, d.chunking_strategy,"
        " COUNT(c.chunk_id) AS n"
        " FROM documents d"
        " LEFT JOIN chunks c USING(doc_id)"
    )
    params: list[str | int] = []
    if lang:
        sql += " WHERE d.lang = ?"
        params.append(lang)
    sql += " GROUP BY d.doc_id ORDER BY d.fetched_at DESC LIMIT ?"
    params.append(limit)
    with SQLiteHelper("rag").open(row_factory=True) as db:
        rows = db.fetchall(sql, tuple(params))
    return [
        {
            "url": r["url"],
            "title": r["title"],
            "lang": r["lang"],
            "fetched_at": r["fetched_at"],
            "chunking_strategy": r["chunking_strategy"],
            "chunk_count": r["n"],
        }
        for r in rows
    ]

async def delete_document(self, url: str) -> bool:
    """Delete a document and its chunks from rag.sqlite by URL."""
    with SQLiteHelper("rag").open(write_mode=True) as db:
        row = db.execute(
            "SELECT doc_id FROM documents WHERE url = ?", (url,)
        ).fetchone()
        if row is None:
            return False
        doc_id = row[0]
        db.execute(
            "DELETE FROM chunks_vec"
            " WHERE chunk_id IN"
            " (SELECT chunk_id FROM chunks WHERE doc_id = ?)",
            (doc_id,),
        )
        db.execute("DELETE FROM chunks WHERE doc_id = ?", (doc_id,))
        db.execute("DELETE FROM documents WHERE doc_id = ?", (doc_id,))
        db.commit()
    return True
```

Add formatter methods:

```python
async def fmt_list_documents(self, args: ToolArgs) -> str:
    """Format rag_list_documents result as text for LLM tool result."""
    lang = args.get("lang")
    limit = int(args["limit"]) if "limit" in args else 20
    rows = await self.list_documents(lang, limit)
    if not rows:
        return "(No documents found.)"
    lines = [f"{r['url']} ({r['lang']}, {r['chunk_count']} chunks)" for r in rows]
    return "\n".join(lines)

async def fmt_delete_document(self, args: ToolArgs) -> str:
    """Format rag_delete_document result as text for LLM tool result."""
    url = args["url"]
    ok = await self.delete_document(url)
    if ok:
        return f"Document deleted: {url}"
    return f"Document not found: {url}"
```

- Thread safety: MCP server is single-process; SQLiteHelper uses per-query connections, safe for concurrent reads
