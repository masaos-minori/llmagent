# Implementation: Error-path tests for file MCP

## Goal

Add missing error-path tests to the 4 existing test files to cover scenarios
identified in the plan (symlink traversal, size limit, grep invalid regex,
atomic write failure, recursive delete root guard, dry-run preview).

## Scope

**Target test files**:
- `tests/test_file_common.py`
- `tests/test_read_service.py`
- `tests/test_file_write_mcp_service.py`
- `tests/test_file_delete_mcp_service.py`

In scope: New test methods only. No changes to existing tests.
Out of scope: New test files.

## Assumptions

1. `test_file_common.py` already has basic `resolve_safe` tests. New tests add
   symlink-escaping-allowed-dir scenario.
2. `test_read_service.py` has a `service` fixture returning `(ReadFileService, root_path)`.
3. `test_file_write_mcp_service.py` has a `service` fixture returning `WriteFileService`.
4. `test_file_delete_mcp_service.py` has a `service` fixture returning `DeleteFileService`.

#### tests/test_file_common.py

Add to `TestSecurityWrappers` or as a new class:

```python
def test_resolve_safe_symlink_escape(self, tmp_path: Path) -> None:
    """Symlink pointing outside allowed_dirs must be rejected."""
    outside = tmp_path / "outside"
    outside.mkdir()
    secret = outside / "secret.txt"
    secret.write_text("secret", encoding="utf-8")

    inside = tmp_path / "workspace"
    inside.mkdir()
    link = inside / "escape_link"
    link.symlink_to(outside)  # link points outside allowed_dirs

    allowed = [inside]
    with pytest.raises(FileAuthorizationError):
        resolve_safe(str(link / "secret.txt"), allowed)
```

def test_grep_files_invalid_regex_raises(self, service: ...) -> None:
    svc, root = service
    req = GrepFilesRequest(path=str(root), pattern="[invalid(regex")
    with pytest.raises(FileValidationError, match="Invalid regular expression"):
        svc.grep_files(req)

#### tests/test_file_write_mcp_service.py

```python
def test_write_file_dry_run_unicode_error_raises(
    self, service: WriteFileService, tmp_path: Path
) -> None:
    """Dry-run on a non-UTF-8 existing file must raise FileValidationError."""
    target = tmp_path / "binary.txt"
    target.write_bytes(b"\xff\xfe bad bytes")
    req = WriteFileRequest(path=str(target), content="new content", dry_run=True)
    with pytest.raises(FileValidationError, match="UTF-8"):
        service.write_file(req)

def test_write_file_content_exceeds_limit(
    self, tmp_path: Path
) -> None:
    """Content exceeding max_write_bytes must raise FileValidationError."""
    svc = WriteFileService(allowed_dirs=[tmp_path], max_write_bytes=10)
    req = WriteFileRequest(path=str(tmp_path / "out.txt"), content="x" * 100)
    with pytest.raises(FileValidationError, match="write limit"):
        svc.write_file(req)
```

#### tests/test_file_delete_mcp_service.py

```python
def test_delete_directory_recursive_root_raises(
    self, tmp_path: Path
) -> None:
    """Recursive delete of an allowed_dir root must raise FileAuthorizationError."""
    svc = DeleteFileService(
        allowed_dirs=[tmp_path],
        audit_log_path=str(tmp_path / "audit.log"),
    )
    (tmp_path / "subdir").mkdir()
    req = DeleteDirectoryRequest(path=str(tmp_path), recursive=True)
    with pytest.raises(FileAuthorizationError, match="allowed root"):
        svc.delete_directory(req)

def test_delete_directory_recursive_subdir_succeeds(
    self, service: DeleteFileService, tmp_path: Path
) -> None:
    """Recursive delete of a subdir must succeed (guard must not block subdirs)."""
    subdir = tmp_path / "toremove"
    subdir.mkdir()
    (subdir / "file.txt").write_text("data", encoding="utf-8")
    req = DeleteDirectoryRequest(path=str(subdir), recursive=True)
    result = service.delete_directory(req)
    assert result.deleted is True
    assert not subdir.exists()
```
