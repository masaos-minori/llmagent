```python
async def _dlq_loop() -> None:
    """Background safety-net sweep for DLQ orphans.

    Under normal operation (inline promotion via nack endpoint working correctly),
    each sweep promotes 0 events. Non-zero sweep count indicates a bug in the
    inline promotion path.
    """
    while True:
        try:
            assert _cfg is not None
            n = await asyncio.to_thread(
                sweep_orphans, _db, _cfg.deadletter_dir, _cfg.max_retry
            )
            if n:
                logger.warning("dlq_loop: swept %d orphan(s) missed by inline promotion", n)
        except Exception:
            logger.exception("dlq_loop error")
        await asyncio.sleep(_DLQ_INTERVAL)
```

Note: added `asyncio.to_thread()` wrapper — `sweep_orphans()` is synchronous and calls SQLite, so it should be offloaded (consistent with req #14 pattern for blocking I/O in async handlers).
