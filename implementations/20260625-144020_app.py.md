**Target `health()`:**
```python
@app.get("/health")
async def health() -> dict[str, Any]:
    import time

    assert _cfg is not None

    def _check_db() -> tuple[str, float, int, int, int, int]:
        t0 = time.monotonic()
        try:
            assert _db is not None
            _db.execute("SELECT 1")
            count_row = _db.execute("SELECT COUNT(*) FROM events").fetchone()
            backlog_row = _db.execute(
                "SELECT COUNT(*) FROM events"
                " WHERE delivery_failure_count > 0 AND dlq_at IS NULL AND acked_at IS NULL"
            ).fetchone()
            orphan_row = _db.execute(
                "SELECT COUNT(*) FROM events"
                " WHERE delivery_failure_count >= ? AND dlq_at IS NULL",
                (_cfg.max_retry,),
            ).fetchone()
            failure_row = _db.execute(
                "SELECT COALESCE(SUM(delivery_failure_count), 0) FROM events"
            ).fetchone()
            latency_ms = (time.monotonic() - t0) * 1000
            return (
                "ok",
                round(latency_ms, 2),
                int(count_row[0]),
                int(backlog_row[0]),
                int(orphan_row[0]),
                int(failure_row[0]),
            )
        except Exception:
            return ("unavailable", -1.0, 0, 0, 0, 0)

    db_status, db_latency_ms, event_count, retry_backlog, dlq_orphan_count, total_delivery_failures = (
        await asyncio.to_thread(_check_db)
    )

    dlq_task_status = (
        "running" if (_dlq_task is not None and not _dlq_task.done()) else "stopped"
    )

    degraded_reasons: list[str] = []
    _RETRY_BACKLOG_THRESHOLD = 10
    if db_status != "ok":
        degraded_reasons.append("db_unavailable")
    if dlq_task_status != "running":
        degraded_reasons.append("dlq_task_stopped")
    if retry_backlog > _RETRY_BACKLOG_THRESHOLD:
        degraded_reasons.append("retry_backlog_high")
    if dlq_orphan_count > 0:
        degraded_reasons.append("dlq_orphan_detected")

    overall = "ok" if not degraded_reasons else "degraded"
    return {
        "status": overall,
        "db": db_status,
        "db_latency_ms": db_latency_ms,
        "dlq_task": dlq_task_status,
        "event_count": event_count,
        "retry_backlog": retry_backlog,
        "dlq_orphan_count": dlq_orphan_count,
        "total_delivery_failures": total_delivery_failures,
        "degraded_reasons": degraded_reasons,
    }
```

Note: `_cfg` is referenced inside the `_check_db` closure. The assertion `assert _cfg is not None` before the `to_thread` call ensures it is set by the time the closure captures it. However, since `_check_db` runs in a thread, use a local variable to avoid type-checking issues:
```python
max_retry = _cfg.max_retry  # capture before to_thread
```
And replace `(_cfg.max_retry,)` with `(max_retry,)` inside the closure.
