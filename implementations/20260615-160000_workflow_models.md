## Goal

Define dataclasses for workflow domain entities: `TaskRecord`, `AttemptRecord`, `ArtifactRef`, `WorkflowDef`.

**Out:**

#### Create `scripts/agent/workflow/models.py`

```python
#!/usr/bin/env python3
"""workflow/models.py — Dataclasses for workflow domain entities."""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Any


class TaskStatus(str, Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    HALTED = "halted"


class AttemptStatus(str, Enum):
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    TIMEOUT = "timeout"


class WorkflowStage(str, Enum):
    PLAN = "plan"
    EXECUTE = "execute"
    VERIFY = "verify"
    RETRY = "retry"


@dataclass(frozen=True)
class TaskRecord:
    """Represents a single workflow task (one agent turn)."""
    task_id: str
    session_id: int
    turn_number: int
    workflow_version: str
    status: TaskStatus = TaskStatus.PENDING
    created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())
    updated_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())

    def mark_running(self) -> TaskRecord:
        return TaskRecord(
            task_id=self.task_id,
            session_id=self.session_id,
            turn_number=self.turn_number,
            workflow_version=self.workflow_version,
            status=TaskStatus.RUNNING,
            created_at=self.created_at,
            updated_at=datetime.utcnow().isoformat(),
        )

    def mark_completed(self) -> TaskRecord:
        return TaskRecord(
            task_id=self.task_id,
            session_id=self.session_id,
            turn_number=self.turn_number,
            workflow_version=self.workflow_version,
            status=TaskStatus.COMPLETED,
            created_at=self.created_at,
            updated_at=datetime.utcnow().isoformat(),
        )

    def mark_failed(self, error: str | None = None) -> TaskRecord:
        return TaskRecord(
            task_id=self.task_id,
            session_id=self.session_id,
            turn_number=self.turn_number,
            workflow_version=self.workflow_version,
            status=TaskStatus.FAILED,
            created_at=self.created_at,
            updated_at=datetime.utcnow().isoformat(),
        )

    def mark_halted(self) -> TaskRecord:
        return TaskRecord(
            task_id=self.task_id,
            session_id=self.session_id,
            turn_number=self.turn_number,
            workflow_version=self.workflow_version,
            status=TaskStatus.HALTED,
            created_at=self.created_at,
            updated_at=datetime.utcnow().isoformat(),
        )


@dataclass(frozen=True)
class AttemptRecord:
    """Represents a single attempt within a task stage."""
    attempt_id: int | None = None
    task_id: str = ""
    stage: WorkflowStage = WorkflowStage.EXECUTE
    status: AttemptStatus = AttemptStatus.RUNNING
    started_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())
    completed_at: str | None = None
    error_message: str | None = None
    attempt_number: int = 1


@dataclass(frozen=True)
class ArtifactRef:
    """Reference to an artifact produced by a workflow stage."""
    artifact_id: str
    task_id: str
    stage_id: str
    uri: str
    created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())


@dataclass(frozen=True)
class RetryPolicy:
    """Retry configuration for a workflow stage."""
    max_attempts: int = 3
    backoff_multiplier: float = 2.0
    retryable_errors: tuple[str, ...] = ()


@dataclass(frozen=True)
class StageDef:
    """Definition of a single workflow stage."""
    id: str
    timeout_sec: int = 60
    retry: RetryPolicy = field(default_factory=RetryPolicy)


@dataclass(frozen=True)
class WorkflowDef:
    """Complete workflow definition loaded from JSON."""
    version: str
    name: str
    stages: list[StageDef]
    global_timeout_sec: int = 300
    on_halt: str = "rollback"

    def get_stage(self, stage_id: str) -> StageDef | None:
        for stage in self.stages:
            if stage.id == stage_id:
                return stage
        return None

    def stage_order(self) -> list[str]:
        return [s.id for s in self.stages]
```

### Details

- `frozen=True` on dataclasses ensures immutability; `mark_*` methods return new instances
- `datetime.utcnow()` used for consistency with SQLite `datetime('now')` format
- `RetryPolicy` and `StageDef` support JSON deserialization from workflow config
- `WorkflowDef.get_stage()` and `stage_order()` provide accessors for stage lookup
