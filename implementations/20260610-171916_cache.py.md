### Method

```python
# rag/cache.py
from __future__ import annotations
import math
from typing import Any, Protocol

from rag.repository import cosine_sim


class CacheService(Protocol):
    def lookup(self, embedding: list[float]) -> str | None: ...
    def put(self, embedding: list[float], context_str: str) -> None: ...


class SemanticCache:
    """In-memory nearest-neighbour cache with dimension validation."""

    def __init__(self, max_size: int = 100, threshold: float = 0.92) -> None:
        self._entries: list[dict[str, Any]] = []
        self._max_size = max_size
        self._threshold = threshold
        self._dim: int | None = None

    def lookup(self, embedding: list[float]) -> str | None:
        if self._dim is not None and len(embedding) != self._dim:
            raise ValueError(
                f"SemanticCache dimension mismatch: expected {self._dim}, got {len(embedding)}"
            )
        best_sim = -1.0
        best_ctx: str | None = None
        for entry in self._entries:
            sim = cosine_sim(embedding, entry["embedding"])
            if sim > best_sim:
                best_sim = sim
                best_ctx = entry["context_str"]
        if best_sim >= self._threshold:
            return best_ctx
        return None

    def put(self, embedding: list[float], context_str: str) -> None:
        if self._dim is None:
            self._dim = len(embedding)
        elif len(embedding) != self._dim:
            raise ValueError(
                f"SemanticCache dimension mismatch: expected {self._dim}, got {len(embedding)}"
            )
        self._entries.append({"embedding": embedding, "context_str": context_str})
        self.prune()

    def prune(self) -> None:
        if len(self._entries) > self._max_size:
            self._entries = self._entries[-self._max_size :]

    @property
    def size(self) -> int:
        return len(self._entries)
```

### Details

- `cosine_sim` is imported from `rag.repository` — this is a downward dependency (`cache` → `repository`) which is fine since `cache` is a leaf module.
- Alternatively, move `cosine_sim` to `rag.utils` to avoid the `cache → repository` dependency entirely. Check import-linter contracts before deciding.
- `deploy/deploy.sh` adds: `cp "$SRC/rag/cache.py" "$DST/rag/"` following the existing pattern.
